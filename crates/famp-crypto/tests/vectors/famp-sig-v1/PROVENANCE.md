import jcs, base64
from cryptography.hazmat.primitives.asymmetric.ed25519 import Ed25519PrivateKey

envelope = {
    "famp": "0.5.1",
    "id": "01890a3b-2c4d-7e5f-8a1b-0c2d3e4f5a6b",
    "from": "agent:example.test/alice",
    "to": "agent:example.test/bob",

    "class": "ack",
    "causality": {"rel": "acknowledges", "ref": "01890a3b-1111-7222-8333-444444444444"},
    "authority": "advisory",
    "ts": "2026-04-13T00:00:00Z",
    "body": {"disposition": "accepted"},
}
canonical = jcs.canonicalize(envelope)                       # bytes
PREFIX = b"FAMP-sig-v1\x00"
signing_input = PREFIX + canonical
sk_hex = "9d61b19deffd5a60ba844af492ec2cc44449c5697b326919703bac031cae7f60"
sk = Ed25519PrivateKey.from_private_bytes(bytes.fromhex(sk_hex))
sig = sk.sign(signing_input)
print("canonical_hex    =", canonical.hex())
print("signing_input    =", signing_input.hex())
