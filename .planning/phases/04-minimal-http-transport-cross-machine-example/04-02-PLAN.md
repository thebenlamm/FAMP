//! Pre-routing signature-verification middleware (D-C1, D-C2, D-C3, TRANS-09).
//!
//! Two-phase decode shape, mirroring `crates/famp/src/runtime/loop_fn.rs`:
//!   1. peek_sender(bytes)            ã extract `from` field strictly
//!   2. keyring.get(&sender)          ã look up pinned key
//!   3. AnySignedEnvelope::decode(&bytes, &key)  ã verify with that exact key
//!
//! Pitfall 3 (RESEARCH.md) ã without two-phase shape, an attacker with ANY
//! pinned key in the keyring could sign for ANY sender. NEVER iterate the
//! keyring trying keys.

use std::{
    sync::Arc,
    task::{Context, Poll},
};

use axum::{
    body::{to_bytes, Body},
    http::Request,
    response::{IntoResponse, Response},
};
use famp_envelope::{peek_sender, AnySignedEnvelope, EnvelopeDecodeError};
use famp_keyring::Keyring;
use futures_util::future::BoxFuture;
use tower::{Layer, Service};

use crate::error::MiddlewareError;

const ONE_MIB: usize = 1_048_576;

#[derive(Clone)]
pub struct FampSigVerifyLayer {
    keyring: Arc<Keyring>,
}

impl FampSigVerifyLayer {
    pub fn new(keyring: Arc<Keyring>) -> Self {
        Self { keyring }
    }
}

impl<S> Layer<S> for FampSigVerifyLayer {
    type Service = FampSigVerifyService<S>;
    fn layer(&self, inner: S) -> Self::Service {
        FampSigVerifyService { inner, keyring: self.keyring.clone() }
    }
}

#[derive(Clone)]
pub struct FampSigVerifyService<S> {
    inner: S,
    keyring: Arc<Keyring>,
}

impl<S> Service<Request<Body>> for FampSigVerifyService<S>
where
    S: Service<Request<Body>, Response = Response, Error = std::convert::Infallible>
        + Clone
        + Send
        + 'static,
    S::Future: Send + 'static,
{
    type Response = Response;
    type Error = std::convert::Infallible;
    type Future = BoxFuture<'static, Result<Response, Self::Error>>;

    fn poll_ready(&mut self, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
        self.inner.poll_ready(cx)
    }

    fn call(&mut self, req: Request<Body>) -> Self::Future {
        // Standard tower clone-and-replace ready-pattern.
        let clone = self.inner.clone();
        let mut inner = std::mem::replace(&mut self.inner, clone);
        let keyring = self.keyring.clone();
        Box::pin(async move {
            let (parts, body) = req.into_parts();
            // Body cap is enforced by the outer RequestBodyLimitLayer; this is a
            // belt-and-suspenders second cap so this layer cannot be the one that
            // pulls unbounded bytes if the outer layer is ever removed.
            let bytes = match to_bytes(body, ONE_MIB).await {
                Ok(b) => b,
                Err(_) => return Ok(MiddlewareError::BodyTooLarge.into_response()),
            };
            // Phase 1: extract sender without verification.
            let sender = match peek_sender(&bytes) {
                Ok(p) => p,
                Err(EnvelopeDecodeError::MalformedJson(_)) => {
                    return Ok(MiddlewareError::BadEnvelope.into_response());
                }
                Err(_) => {
                    return Ok(MiddlewareError::BadEnvelope.into_response());
                }
            };
            // Phase 2: keyring lookup against the claimed sender.
            let key = match keyring.get(&sender) {
                Some(k) => k.clone(),
                None => return Ok(MiddlewareError::UnknownSender.into_response()),
            };
            // Phase 3: full decode + verify-against-pinned-key.
            // Use whatever sister API famp-envelope exposes that takes a pinned key.
            // If the only API is `AnySignedEnvelope::decode(&bytes, &key)`, call that.
            // Map the typed errors to the LOAD-BEARING D-C6 status codes.
            let envelope = match AnySignedEnvelope::decode(&bytes, &key) {
                Ok(e) => e,
                Err(EnvelopeDecodeError::SignatureInvalid) => {
                    return Ok(MiddlewareError::SignatureInvalid.into_response());
                }
                Err(EnvelopeDecodeError::MissingSignature) => {
                    return Ok(MiddlewareError::BadEnvelope.into_response());
                }
                Err(EnvelopeDecodeError::CanonicalDivergence) => {
                    return Ok(MiddlewareError::CanonicalDivergence.into_response());
                }
                Err(_) => return Ok(MiddlewareError::BadEnvelope.into_response()),
            };
            // Reattach body. Stash decoded envelope (D-C3, informational).
            let mut req = Request::from_parts(parts, Body::from(bytes));
            req.extensions_mut().insert(Arc::new(envelope));
            inner.call(req).await
        })
    }
}
```

//! axum 0.8 router + inbox handler for `famp-transport-http`.
//!
//! D-A1: ONE listener per process, path-multiplexed via `POST /famp/v0.5.1/inbox/{principal}`.
//! D-C1: tower layer order (outer -> inner): RequestBodyLimitLayer -> FampSigVerifyLayer -> handler.
//! D-A3: handler forwards RAW body bytes into the per-principal mpsc inbox; runtime
//!       glue re-decodes (deliberate double-check, see CONTEXT.md D-C2 reasoning).

use std::{collections::HashMap, str::FromStr, sync::Arc};

use axum::{
    body::Bytes,
    extract::{Path, State},
    http::StatusCode,
    routing::post,
    Router,
};
use famp_core::Principal;
use famp_keyring::Keyring;
use famp_transport::TransportMessage;
use tokio::sync::{mpsc, Mutex};
use tower::ServiceBuilder;
use tower_http::limit::RequestBodyLimitLayer;

use crate::{error::MiddlewareError, middleware::FampSigVerifyLayer};

const ONE_MIB: usize = 1_048_576;
pub const INBOX_ROUTE: &str = "/famp/v0.5.1/inbox/{principal}";

pub type InboxRegistry = Mutex<HashMap<Principal, mpsc::Sender<TransportMessage>>>;

#[derive(Clone)]
pub struct ServerState {
    pub inboxes: Arc<InboxRegistry>,
}

pub fn build_router(keyring: Arc<Keyring>, inboxes: Arc<InboxRegistry>) -> Router {
    let state = ServerState { inboxes };
    Router::new()
        .route(INBOX_ROUTE, post(inbox_handler))
        .with_state(state)
        .layer(
            ServiceBuilder::new()
                // OUTER ã runs first per Pitfall 2. 1 MB cap per spec ôÏ18 (TRANS-07).
                .layer(RequestBodyLimitLayer::new(ONE_MIB))
                // INNER ã runs second. Decode + verify; rejects before route dispatch.
                .layer(FampSigVerifyLayer::new(keyring)),
        )
}

async fn inbox_handler(
    Path(principal_str): Path<String>,
    State(state): State<ServerState>,
    body: Bytes,
) -> Result<StatusCode, MiddlewareError> {
    let recipient = Principal::from_str(&principal_str)
        .map_err(|_| MiddlewareError::BadPrincipal)?;
    // D-A3 step 5: forward raw bytes (sender not consulted here ã runtime re-decodes
    // to fill the TransportMessage.sender field via its own peek_sender call).
    let inboxes = state.inboxes.lock().await;
    let sender = inboxes
        .get(&recipient)
        .ok_or(MiddlewareError::UnknownRecipient)?
        .clone();
    drop(inboxes);
    sender
        .send(TransportMessage {
            // The middleware stashed Arc<AnySignedEnvelope> in extensions, but D-A3
            // says forward raw bytes; runtime re-derives sender. To avoid a double
            // peek_sender call here, set sender = recipient as a placeholder ONLY
            // if the runtime contract requires it. Confirm against
            // crates/famp/src/runtime/loop_fn.rs - Phase 3 D-D5 already passes raw
            // bytes via MemoryTransport; mirror that exact shape.
            sender: recipient.clone(),
            recipient: recipient.clone(),
            bytes: body.to_vec(),
        })
        .await
        .map_err(|_| MiddlewareError::Internal)?;
    Ok(StatusCode::ACCEPTED)
}
```

//! requests BEFORE the handler closure is invoked.
//!

#![allow(clippy::unwrap_used, unused_crate_dependencies)]

use std::sync::{
    atomic::{AtomicBool, Ordering},
    Arc,
};

use axum::{body::Body, http::Request, http::StatusCode, routing::post, Router};
use famp_keyring::Keyring;
use famp_transport_http::middleware::FampSigVerifyLayer;
use tower::{ServiceBuilder, ServiceExt};
use tower_http::limit::RequestBodyLimitLayer;

const ONE_MIB: usize = 1_048_576;

fn sentinel_router(keyring: Arc<Keyring>, sentinel: Arc<AtomicBool>) -> Router {
    Router::new()
        .route(
            "/famp/v0.5.1/inbox/{principal}",
            post({
                let s = sentinel.clone();
                move || {
                    let s = s.clone();
                    async move {
                        s.store(true, Ordering::SeqCst);
                        StatusCode::ACCEPTED
                    }
                }
            }),
        )
        .layer(
            ServiceBuilder::new()
                .layer(RequestBodyLimitLayer::new(ONE_MIB))
                .layer(FampSigVerifyLayer::new(keyring)),
        )
}

fn empty_keyring() -> Arc<Keyring> {
    Arc::new(Keyring::default())
}

#[tokio::test]
async fn unsigned_request_does_not_enter_handler() {
    let sentinel = Arc::new(AtomicBool::new(false));
    let app = sentinel_router(empty_keyring(), sentinel.clone());
    let req = Request::builder()
        .method("POST")
        .uri("/famp/v0.5.1/inbox/agent:local/bob")
        .body(Body::from(br#"{"from":"agent:local/alice"}"#.to_vec()))
        .unwrap();

    assert!(resp.status().is_client_error(), "should reject");
    assert!(!sentinel.load(Ordering::SeqCst), "TRANS-09 SC#2: handler must NOT be entered");
}
