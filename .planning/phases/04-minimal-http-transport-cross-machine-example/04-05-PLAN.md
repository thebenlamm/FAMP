Wait — `tests/` integration tests in Cargo work where each `.rs` file at the top of `tests/` is its own test binary. We CANNOT have BOTH `tests/adversarial.rs` AND `tests/adversarial/mod.rs` directly compete. The clean pattern is:

- Keep ONE entry file: `tests/adversarial.rs`
- Inside it: `mod harness; mod memory; mod http; mod fixtures;` where `harness.rs` etc. live in a sibling subdirectory `tests/adversarial/`

Cargo treats `tests/adversarial.rs` as a single test binary; modules referenced from it via `mod foo` resolve via `tests/adversarial/foo.rs` if you use `#[path = "adversarial/foo.rs"] mod foo;` OR by placing them in a directory named after the file. **The simplest layout that actually compiles:**

```
tests/adversarial.rs                   # single test binary, declares all modules
tests/adversarial/mod.rs               # NOT used by Cargo — delete this idea
```

Use explicit `#[path = ...]` to keep submodule files in `tests/adversarial/`:

```rust
// tests/adversarial.rs
#![allow(clippy::unwrap_used, unused_crate_dependencies)]

#[path = "adversarial/harness.rs"]   mod harness;
#[path = "adversarial/fixtures.rs"]  mod fixtures;

//! Shared adversarial harness — Phase 4 D-D2/D-D3.
//!
//! ONE definition of the three cases, ONE expected-error map, ONE assertion
//! shape. Each transport adapter implements `AdversarialTransport`; each test
//! row calls the same `assert_expected_error`.

use famp::runtime::error::RuntimeError;  // adjust path if RuntimeError is re-exported elsewhere
use famp_core::Principal;
use famp_envelope::EnvelopeDecodeError;

#[derive(Debug, Clone, Copy)]
pub enum Case {
    Unsigned,
    WrongKey,
    CanonicalDivergence,
}

pub trait AdversarialTransport {
    /// Inject raw wire bytes claiming `sender → recipient`. The transport adapter
    /// is responsible for routing them through the same path a real message would
    /// take, including any middleware. Returns nothing — the side-effect is that
    /// `run_loop_and_catch` will observe (or not observe) the resulting error.
    async fn inject_raw(&self, sender: &Principal, recipient: &Principal, bytes: &[u8]);

    /// Drive the receive loop for `as_principal` and return whatever `RuntimeError`
    /// surfaces. Each adapter is responsible for waiting long enough that the
    /// inject_raw delivery has completed.
    async fn run_loop_and_catch(&self, as_principal: &Principal) -> RuntimeError;
}

/// D-D6 expected-error mapping (RuntimeError side — same on both transports).
pub fn assert_expected_error(case: Case, err: &RuntimeError) {
    match (case, err) {
        (Case::Unsigned, RuntimeError::Decode(EnvelopeDecodeError::MissingSignature)) => {}
        (Case::WrongKey, RuntimeError::Decode(EnvelopeDecodeError::SignatureInvalid)) => {}
        (Case::CanonicalDivergence, RuntimeError::CanonicalDivergence) => {}
        _ => panic!("case {:?} produced unexpected error: {:?}", case, err),
    }
}
```

```rust
//! HttpTransport adversarial rows + sentinel proof of TRANS-09 SC#2.
//!
//! D-D2: raw reqwest POST as the injection surface (no test-util feature).
//! D-D5: Arc<AtomicBool> sentinel asserts the handler closure is NOT entered
//!       on any of the three adversarial cases.
//! D-D6 HTTP column: the server response status + body slug must match the
//!       Phase 4 D-C6 mapping per case.

use super::{
    fixtures::case_bytes,
    harness::{assert_expected_error, AdversarialTransport, Case},
};
use famp_core::Principal;
use famp_keyring::Keyring;
use famp::runtime::error::RuntimeError;
use famp_transport_http::{build_router, server::InboxRegistry};
use std::{
    collections::HashMap,
    str::FromStr,
    sync::{
        atomic::{AtomicBool, Ordering},
        Arc,
    },
};
use tokio::{net::TcpListener, sync::Mutex};

struct HttpAdapter {
    base_url: String,
    sentinel: Arc<AtomicBool>,
    last_status: Arc<Mutex<Option<u16>>>,
    last_slug: Arc<Mutex<Option<String>>>,
}

async fn build_test_server(keyring: Arc<Keyring>) -> (HttpAdapter, tokio::task::JoinHandle<()>) {
    // Mount the REAL build_router so we exercise the real middleware chain,
    // then wrap the resulting Router in a sentinel-recording layer that
    // observes whether the handler was reached.
    let inboxes: Arc<InboxRegistry> = Arc::new(Mutex::new(HashMap::new()));
    // Pre-register agent:local/bob so unknown_recipient is NOT the failure mode.
    let bob = Principal::from_str("agent:local/bob").unwrap();
    let (tx, _rx) = tokio::sync::mpsc::channel(8);
    inboxes.lock().await.insert(bob, tx);

    let sentinel = Arc::new(AtomicBool::new(false));
    let sentinel_for_router = sentinel.clone();

    // The router under test is build_router(...) but we insert a from_fn map
    // AFTER all internal layers (i.e., as the innermost wrapper around the
    // actual handler) that flips the sentinel. Because we cannot easily inject
    // into build_router's internal handler, the cleanest approach is:
    //   - call build_router to build the canonical Router
    //   - .layer() a `tower::service_fn`-style observation layer that triggers
    //     ONLY when the inner handler returns (i.e., handler-was-entered)
    // tower-http does not ship this directly; the simplest impl is a custom
    // tower::Layer that flips an AtomicBool BEFORE calling inner.call() — but
    // .layer() runs OUTERMOST, so this would flip on every request including
    // ones the existing layers reject. NOT WHAT WE WANT.
    //
    // Correct approach: use `Router::route_layer` to attach a layer that runs
    // AFTER routing but BEFORE the handler closure. axum 0.8 supports this via
    // `route_layer`, which is exactly the seam we need.
    //
    // Implementation:
    let router = build_router(keyring, inboxes.clone());
    let router = router.route_layer(axum::middleware::from_fn(move |req, next| {
        let s = sentinel_for_router.clone();
        async move {
            s.store(true, Ordering::SeqCst);
            next.run(req).await
        }
    }));

    let listener = TcpListener::bind("127.0.0.1:0").await.unwrap();
    let addr = listener.local_addr().unwrap();
    let handle = tokio::spawn(async move {
        axum::serve(listener, router).await.unwrap();
    });
    let adapter = HttpAdapter {
        base_url: format!("http://{}", addr),  // plain HTTP for the adversarial rig — TLS adds no value here
        sentinel,
        last_status: Arc::new(Mutex::new(None)),
        last_slug: Arc::new(Mutex::new(None)),
    };
    (adapter, handle)
}

impl AdversarialTransport for HttpAdapter {
    async fn inject_raw(&self, _sender: &Principal, recipient: &Principal, bytes: &[u8]) {
        let url = format!("{}/famp/v0.5.1/inbox/{}", self.base_url, recipient);
        let resp = reqwest::Client::new()
            .post(&url)
            .header("content-type", "application/famp+json")
            .body(bytes.to_vec())
            .send()
            .await
            .expect("inject_raw send");
        let status = resp.status().as_u16();
        let body = resp.text().await.unwrap_or_default();
        let slug = serde_json::from_str::<serde_json::Value>(&body)
            .ok()
            .and_then(|v| v.get("error").and_then(|e| e.as_str().map(String::from)));
        *self.last_status.lock().await = Some(status);
        *self.last_slug.lock().await = slug;
    }

    async fn run_loop_and_catch(&self, _as_principal: &Principal) -> RuntimeError {
        // For HTTP adversarial rows the runtime loop is NOT driven (the message
        // never lands in the inbox channel because middleware rejects it). We
        // synthesize a RuntimeError from the recorded HTTP status+slug so the
        // shared assert_expected_error works uniformly. This is the one place
        // the harness intentionally diverges between transports.
        let status = self.last_status.lock().await.unwrap_or(0);
        let slug = self.last_slug.lock().await.clone().unwrap_or_default();
        match (status, slug.as_str()) {
            (400, "bad_envelope")         => RuntimeError::Decode(famp_envelope::EnvelopeDecodeError::MissingSignature),
            (401, "signature_invalid")    => RuntimeError::Decode(famp_envelope::EnvelopeDecodeError::SignatureInvalid),
            (400, "canonical_divergence") => RuntimeError::CanonicalDivergence,
            _ => panic!("HTTP adversarial returned status={} slug={} (not in expected mapping)", status, slug),
        }
    }
}

async fn run_http_case(case: Case) {
    let keyring = Arc::new(build_test_keyring());  // alice + bob both pinned correctly for the wrong-key case
    let (adapter, handle) = build_test_server(keyring).await;
    let alice = Principal::from_str("agent:local/alice").unwrap();
    let bob = Principal::from_str("agent:local/bob").unwrap();
    adapter.inject_raw(&alice, &bob, &case_bytes(case)).await;
    let err = adapter.run_loop_and_catch(&bob).await;
    assert_expected_error(case, &err);
    // D-D5 sentinel proof — the SAME assertion for ALL three cases.
    assert!(
        !adapter.sentinel.load(Ordering::SeqCst),
        "TRANS-09 SC#2: handler closure entered on adversarial case {:?}",
        case
    );
    handle.abort();
}

fn build_test_keyring() -> Keyring {
    // alice's pubkey pinned to alice's principal, bob's pubkey pinned to bob's principal.
    // Reuse the Phase 3 helper if one exists in tests/adversarial/memory.rs or as a shared fn.
    todo!("mirror Phase 3 keyring construction")
}

#[tokio::test] async fn http_unsigned()              { run_http_case(Case::Unsigned).await; }
#[tokio::test] async fn http_wrong_key()             { run_http_case(Case::WrongKey).await; }
#[tokio::test] async fn http_canonical_divergence()  { run_http_case(Case::CanonicalDivergence).await; }
```
