```rust
//! Shared happy-path cycle driver — extracted from
//! `crates/famp/examples/personal_two_agents.rs` in Phase 4 Plan 04-04 so both
//! the same-process HTTP safety-net test and the HTTP example can call one
//! canonical implementation of the alice/bob halves of the signed cycle.
//!
//! Generic over `T: Transport`, so the same driver runs on `MemoryTransport`
//! (Phase 3) or `HttpTransport` (Phase 4).

#![allow(clippy::unwrap_used, clippy::expect_used, clippy::too_many_arguments, dead_code)]

use famp::runtime::{adapter, process_one_message, RuntimeError};
use famp_core::{AuthorityScope, MessageId, Principal};
use famp_crypto::FampSigningKey;
use famp_envelope::body::{
    AckBody, AckDisposition, Bounds, Budget, CommitBody, DeliverBody, RequestBody, TerminalStatus,
};
use famp_envelope::{AnySignedEnvelope, SignedEnvelope, Timestamp, UnsignedEnvelope};
use famp_fsm::TaskFsm;
use famp_keyring::Keyring;
use famp_transport::{Transport, TransportMessage};
use std::sync::{Arc, Mutex};

pub type Trace = Arc<Mutex<Vec<String>>>;

pub fn ts() -> Timestamp { Timestamp("2026-04-13T00:00:00Z".to_string()) }

pub fn two_key_bounds() -> Bounds {
    Bounds {
        deadline: Some("2026-05-01T00:00:00Z".to_string()),
        budget: Some(Budget { amount: "100".to_string(), unit: "usd".to_string() }),
        hop_limit: None, policy_domain: None, authority_scope: None,
        max_artifact_size: None, confidence_floor: None, recursion_depth: None,
    }
}

pub fn canonical_bytes<B: famp_envelope::BodySchema>(signed: &SignedEnvelope<B>) -> Vec<u8> {
    let encoded = signed.encode().expect("encode");
    let value: serde_json::Value = famp_canonical::from_slice_strict(&encoded).expect("strict");
    famp_canonical::canonicalize(&value).expect("canonicalize")
}

pub fn log_line(trace: &Trace, line: String) {
    println!("{line}");
    trace.lock().unwrap().push(line);
}

async fn send_signed<T: Transport, B: famp_envelope::BodySchema>(
    transport: &T,
    sender: &Principal,
    recipient: &Principal,
    signed: &SignedEnvelope<B>,
) where T::Error: std::fmt::Debug {
    let bytes = canonical_bytes(signed);
    let msg = TransportMessage {
        sender: sender.clone(),
        recipient: recipient.clone(),
        bytes,
    };
    transport.send(msg).await.expect("transport send");
}

pub async fn drive_bob<T: Transport>(
    transport: &T,
    bob_keyring: &Keyring,
    bob: &Principal,
    alice: &Principal,
    bob_sk: &FampSigningKey,
    trace: &Trace,
) -> Result<(), RuntimeError>
where T::Error: std::fmt::Debug + Send + Sync + 'static
{
    let mut fsm = TaskFsm::new();

    // 1. Receive request.
    let req_msg = transport.recv(bob).await
        .map_err(|e| RuntimeError::Transport(Box::new(format!("{e:?}").to_string().into()) as _))
        .unwrap_or_else(|_| panic!("recv request"));
    let req_env = process_one_message(&req_msg, bob_keyring, &mut fsm)?;
    assert!(matches!(req_env, AnySignedEnvelope::Request(_)));
    log_line(trace, format!("[1] {} -> {}: Request", adapter::envelope_sender(&req_env), bob));

    // 2. Send commit.
    let commit_body = CommitBody {
        scope: serde_json::json!({"task": "translate"}),
        scope_subset: None,
        bounds: two_key_bounds(),
        accepted_policies: vec!["policy://famp/v0.7/personal".to_string()],
        delegation_permissions: None,
        reporting_obligations: None,
        terminal_condition: serde_json::json!({"type": "final_delivery"}),
        conditions: None,
        natural_language_summary: None,
    };
    let commit = UnsignedEnvelope::<CommitBody>::new(
        MessageId::new_v7(), bob.clone(), alice.clone(),
        AuthorityScope::CommitLocal, ts(), commit_body,
    ).sign(bob_sk).expect("sign commit");
    send_signed(transport, bob, alice, &commit).await;

    // 3. Send deliver (terminal=Completed).
    let deliver_body = DeliverBody {
        interim: false,
        artifacts: None,
        result: Some(serde_json::json!({"text": "Bonjour le monde."})),
        usage_metrics: None,
        error_detail: None,
        provenance: Some(serde_json::json!({"signer": "agent:local/bob"})),
        natural_language_summary: None,
    };
    let deliver = UnsignedEnvelope::<DeliverBody>::new(
        MessageId::new_v7(), bob.clone(), alice.clone(),
        AuthorityScope::Advisory, ts(), deliver_body,
    )
    .with_terminal_status(TerminalStatus::Completed)
    .sign(bob_sk)
    .expect("sign deliver");
    send_signed(transport, bob, alice, &deliver).await;

    // 4. Receive ack from alice.
    let ack_msg = transport.recv(bob).await
        .unwrap_or_else(|e| panic!("recv ack: {e:?}"));
    let ack_env = process_one_message(&ack_msg, bob_keyring, &mut fsm)?;
    assert!(matches!(ack_env, AnySignedEnvelope::Ack(_)));
    Ok(())
}

pub async fn drive_alice<T: Transport>(
    transport: &T,
    alice_keyring: &Keyring,
    alice: &Principal,
    bob: &Principal,
    alice_sk: &FampSigningKey,
    trace: &Trace,
) -> Result<(), RuntimeError>
where T::Error: std::fmt::Debug + Send + Sync + 'static
{
    let mut fsm = TaskFsm::new();

    // 1. Send request.
    let req_body = RequestBody {
        scope: serde_json::json!({"task": "translate"}),
        bounds: two_key_bounds(),
        natural_language_summary: Some("translate to french".to_string()),
    };
    let req = UnsignedEnvelope::<RequestBody>::new(
        MessageId::new_v7(), alice.clone(), bob.clone(),
        AuthorityScope::Advisory, ts(), req_body,
    ).sign(alice_sk).expect("sign request");
    send_signed(transport, alice, bob, &req).await;

    // 2. Receive commit.
    let commit_msg = transport.recv(alice).await.unwrap_or_else(|e| panic!("recv commit: {e:?}"));
    let commit_env = process_one_message(&commit_msg, alice_keyring, &mut fsm)?;
    assert!(matches!(commit_env, AnySignedEnvelope::Commit(_)));
    log_line(trace, format!("[2] {} -> {}: Commit", adapter::envelope_sender(&commit_env), alice));

    // 3. Receive deliver.
    let deliver_msg = transport.recv(alice).await.unwrap_or_else(|e| panic!("recv deliver: {e:?}"));
    let deliver_env = process_one_message(&deliver_msg, alice_keyring, &mut fsm)?;
    assert!(matches!(deliver_env, AnySignedEnvelope::Deliver(_)));
    log_line(trace, format!("[3] {} -> {}: Deliver", adapter::envelope_sender(&deliver_env), alice));

    // 4. Send ack.
    let ack_body = AckBody { disposition: AckDisposition::Completed, reason: None };
    let ack = UnsignedEnvelope::<AckBody>::new(
        MessageId::new_v7(), alice.clone(), bob.clone(),
        AuthorityScope::Advisory, ts(), ack_body,
    ).sign(alice_sk).expect("sign ack");
    send_signed(transport, alice, bob, &ack).await;
    log_line(trace, format!("[4] {alice} -> {bob}: Ack"));
    Ok(())
}
```

Create `crates/famp/examples/cross_machine_two_agents.rs`. The binary:
- `--role alice|bob` — fixed principal + default port (alice 8444, bob 8443) (D-E2)
- `--listen <addr:port>` — override, `127.0.0.1:0` for ephemeral (D-E6)
- `--peer <principal>=<base64url-pubkey>` — pin peer pubkey
- `--addr <principal>=<https-url>` — where to POST to that peer
- `--cert <path>` `--key <path>` — server cert/key (D-B5); if absent, generate via rcgen
- `--trust-cert <path>` — client trust anchor added to OS root store via D-B5 `build_client_config`
- `--out-pubkey <path>` `--out-cert <path>` `--out-key <path>` — dump self pubkey/cert/key for the peer to copy

Binary flow:
  1. parse args
  2. generate Ed25519 signing key; write pubkey (base64url) to `--out-pubkey` if specified
  3. generate or load TLS cert/key; write PEMs to `--out-cert` / `--out-key` if specified (D-B7 path)
  4. build `Keyring` with peers pinned (own key + peers from `--peer`)
  5. build `HttpTransport::new_client_only(Some(trust_cert))` — NOTE: trust_cert is the PEER's cert file, not self
  6. `transport.register(me).await`
  7. for each `(p, url)` in `--addr`: `transport.add_peer(p, url).await`
  8. build `tls::build_server_config(load_pem_cert(cert), load_pem_key(key))`
  9. build `build_router(Arc::new(keyring), transport.inboxes())`
  10. bind `std::net::TcpListener::bind(args.listen)`, read `local_addr()`, **print `LISTENING https://{addr}`** to stderr (subprocess-sync beacon)
  11. call `tls_server::serve_std_listener(listener, router, Arc::new(server_config))` → JoinHandle
  12. `transport.attach_server(handle).await`
  13. dispatch to `cycle_driver::drive_alice(...)` or `drive_bob(...)` based on `--role`

```rust
//! ports, tempdir cert exchange. Both procs must exit 0 within 15 s.

use std::{

use tempfile::tempdir;
use tokio::{
    io::{AsyncBufReadExt, BufReader},
    process::Command,
    time::timeout,

#[tokio::test]
async fn cross_machine_request_commit_deliver_ack() {
    let dir = tempdir().unwrap();
    let bob_pub  = dir.path().join("bob.pub");
    let bob_crt  = dir.path().join("bob.crt");
    let bob_key  = dir.path().join("bob.key");
    let alice_pub = dir.path().join("alice.pub");
    let alice_crt = dir.path().join("alice.crt");
    let alice_key = dir.path().join("alice.key");

    // Step 1: spawn bob on ephemeral port.
    let mut bob = Command::new(env!("CARGO_BIN_EXE_cross_machine_two_agents"))
        .args([
            "--role", "bob",
            "--listen", "127.0.0.1:0",
            "--out-pubkey", bob_pub.to_str().unwrap(),
            "--out-cert",   bob_crt.to_str().unwrap(),
            "--out-key",    bob_key.to_str().unwrap(),
        ])
        .stdout(Stdio::piped())
        .stderr(Stdio::piped())
        .kill_on_drop(true)
        .spawn()
        .expect("spawn bob");

    // Step 2: scan bob's stderr for "LISTENING https://<addr>".
    let bob_stderr = bob.stderr.take().expect("bob stderr");
    let mut lines = BufReader::new(bob_stderr).lines();
    let mut bob_addr = String::new();
    let deadline = Instant::now() + Duration::from_secs(5);
    while Instant::now() < deadline {
        match timeout(Duration::from_millis(500), lines.next_line()).await {
            Ok(Ok(Some(line))) => {
                if let Some(rest) = line.strip_prefix("LISTENING https://") {
                    bob_addr = rest.trim().to_string();
                    break;
                }
            }
            _ => {}
        }
    }
    assert!(!bob_addr.is_empty(), "bob did not print LISTENING before 5s");

    // Step 3: wait for bob's pubkey + cert files.
    let deadline = Instant::now() + Duration::from_secs(3);
    while Instant::now() < deadline && !(bob_pub.exists() && bob_crt.exists()) {
        tokio::time::sleep(Duration::from_millis(50)).await;
    }
    assert!(bob_pub.exists() && bob_crt.exists(), "bob did not write pub/cert files");
    let bob_pub_b64 = std::fs::read_to_string(&bob_pub).unwrap().trim().to_string();

    // Step 4: spawn alice pointed at bob.
    let alice = Command::new(env!("CARGO_BIN_EXE_cross_machine_two_agents"))
        .args([
            "--role", "alice",
            "--listen", "127.0.0.1:0",
            "--peer",       &format!("agent:local/bob={}", bob_pub_b64),
            "--addr",       &format!("agent:local/bob=https://{}", bob_addr),
            "--trust-cert", bob_crt.to_str().unwrap(),
            "--out-pubkey", alice_pub.to_str().unwrap(),
            "--out-cert",   alice_crt.to_str().unwrap(),
            "--out-key",    alice_key.to_str().unwrap(),

        .stderr(Stdio::piped())
```

**Caveat:** This subprocess test assumes alice-side also receives bob's commit/deliver. For that, alice needs to bind her own listener AND bob needs `--peer agent:local/alice=...` + `--addr agent:local/alice=https://...`. The current CLI shape doesn't pass alice's info to bob at spawn time because bob has to start first. Resolution: bob runs in "accept-any-peer" mode by keeping `--peer` empty at startup and relying on alice's request arrival (bob's keyring is initially empty, so alice's request → `UnknownSender` → bob fails).

**The clean answer:** For the subprocess test, have alice start FIRST with `--listen` but NOT yet connecting, print her own LISTENING + write pub/cert, then test code spawns bob with full `--peer agent:local/alice=...` + `--addr`, bob prints LISTENING, test code then SIGNALs alice (via another file watch or stdin) to start sending. This is complex.

**Alternative: three-pass startup.** The simplest path is to invert the example binary: `--role bob` writes its pubkey/cert and WAITS for a `ready.signal` file (which the test creates after writing alice's pub/cert to disk). Then bob reads alice's pub/cert, loads them, and only THEN starts the cycle. Meanwhile alice starts with bob's info already on disk.

**Pragmatic resolution for Plan 04-04:** If the executor finds the subprocess orchestration too fragile, ship the same-process test (below) as the CONF-04 gate and mark the subprocess test `#[ignore]` with a comment explaining the bootstrap problem. Document the decision. The same-process test exercises the SAME axum router + real rustls TLS (we can make it bind to 127.0.0.1:0 with real certs), just without process isolation. This is acceptable because CONF-04 asks for the cycle over HttpTransport, not specifically across OS processes.
