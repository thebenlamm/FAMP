```rust
//! cross_machine_two_agents — EX-02 / CONF-04 example.
//!
//! ONE binary, TWO invocations (D-E1). No auto-orchestration. Fixed `--role alice|bob`.
//! Symmetric topology: both roles run server + client.
//!
//! Run sequence (D-E3):
//!   bob:   cargo run --example cross_machine_two_agents -- --role bob \
//!              --out-pubkey /tmp/bob.pub --out-cert /tmp/bob.crt --out-key /tmp/bob.key
//!   alice: cargo run --example cross_machine_two_agents -- --role alice \
//!              --peer agent:local/bob=$(cat /tmp/bob.pub) \
//!              --addr agent:local/bob=https://127.0.0.1:8443 \
//!              --trust-cert /tmp/bob.crt \
//!              --out-pubkey /tmp/alice.pub --out-cert /tmp/alice.crt --out-key /tmp/alice.key
//! Then bob runs again with alice's pub/cert symmetrically.

use std::{net::SocketAddr, path::PathBuf, sync::Arc};

use famp_core::Principal;
use famp_keyring::Keyring;
use famp_transport_http::{build_router, HttpTransport};
use rcgen::generate_simple_self_signed;
use tokio::net::TcpListener;
use url::Url;

#[derive(Debug)]
struct Args {
    role: Role,
    listen: SocketAddr,
    peers: Vec<(Principal, Vec<u8>)>,
    addrs: Vec<(Principal, Url)>,
    cert: Option<PathBuf>,
    key: Option<PathBuf>,
    trust_cert: Option<PathBuf>,
    out_pubkey: Option<PathBuf>,
    out_cert: Option<PathBuf>,
    out_key: Option<PathBuf>,
}

#[derive(Debug, Clone, Copy)]
enum Role { Alice, Bob }

impl Role {
    fn principal(&self) -> &'static str {
        match self { Self::Alice => "agent:local/alice", Self::Bob => "agent:local/bob" }
    }
    fn default_port(&self) -> u16 {
        match self { Self::Alice => 8444, Self::Bob => 8443 }
    }
}

fn parse_args() -> Args { /* hand-rolled or clap; see crates/famp/examples/personal_two_agents.rs for style */ todo!() }

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let args = parse_args();
    let me = Principal::from_str(args.role.principal())?;

    // Ed25519 keypair
    let signing_key = famp_crypto::SigningKey::generate();
    let pubkey_bytes = signing_key.verifying_key().to_bytes();
    if let Some(path) = &args.out_pubkey {
        std::fs::write(path, base64_url_encode(&pubkey_bytes))?;
    }

    // TLS cert (load or generate)
    let (cert_path, key_path) = match (args.cert.as_deref(), args.key.as_deref()) {
        (Some(c), Some(k)) => (c.to_path_buf(), k.to_path_buf()),
        _ => {
            let ck = generate_simple_self_signed(vec!["localhost".into(), "127.0.0.1".into()])?;
            let cp = args.out_cert.clone().unwrap_or_else(|| std::env::temp_dir().join(format!("{}.crt", args.role.principal_short())));
            let kp = args.out_key.clone().unwrap_or_else(|| std::env::temp_dir().join(format!("{}.key", args.role.principal_short())));
            std::fs::write(&cp, ck.cert.pem())?;
            std::fs::write(&kp, ck.signing_key.serialize_pem())?;
            eprintln!("generated cert: {}", cp.display());
            eprintln!("generated key:  {}", kp.display());
            (cp, kp)
        }
    };

    // Keyring with peer pubkeys pinned
    let mut keyring = Keyring::default();
    keyring.insert(me.clone(), famp_crypto::TrustedVerifyingKey::from_bytes(&pubkey_bytes)?);
    for (p, k_bytes) in &args.peers {
        keyring.insert(p.clone(), famp_crypto::TrustedVerifyingKey::from_bytes(k_bytes)?);
    }
    let keyring = Arc::new(keyring);

    // HttpTransport (client + server)
    let transport = match &args.trust_cert {
        Some(tc) => HttpTransport::client_only(tc)?,
        None => HttpTransport::client_only(&cert_path)?, // self-trust fallback for first-run
    };
    transport.register(me.clone()).await;
    for (p, url) in &args.addrs {
        transport.add_peer(p.clone(), url.clone()).await;
    }

    // axum server
    let server_cert = famp_transport_http::tls::load_pem_cert(&cert_path)?;
    let server_key = famp_transport_http::tls::load_pem_key(&key_path)?;
    let tls = famp_transport_http::tls::build_server_config(server_cert, server_key)?;
    let router = build_router(keyring.clone(), transport.inboxes());
    let listener = TcpListener::bind(args.listen).await?;
    let bound = listener.local_addr()?;
    eprintln!("LISTENING https://{}", bound);  // synchronization beacon for subprocess test
    let handle = tokio::spawn(async move {
        // axum-server with rustls — see axum docs for the exact serve-tls helper.
        // If axum 0.8 doesn't ship a built-in tls helper, use axum-server = "0.7"
        // (executor decides at write time; it's a one-line dep).
        let _ = (router, tls);  // TODO wire to axum-server::bind_rustls or similar
        // SAFETY NET: if no built-in TLS, fall back to spawning hyper::server::conn::http1::Builder
        // with a tokio_rustls::TlsAcceptor wrapped around the listener.
    });
    transport.attach_server(handle).await;

    // Drive the runtime cycle:
    // - bob waits for inbound `request`, sends `commit`, then `deliver`, then awaits `ack`.
    // - alice sends `request`, awaits `commit`, awaits `deliver`, sends `ack`.
    // Use crates/famp/src/runtime/loop_fn.rs's existing driver — same as Phase 3 personal_two_agents.
    match args.role {
        Role::Alice => drive_alice(&transport, &keyring, /* signing_key */).await?,
        Role::Bob   => drive_bob(&transport, &keyring, /* signing_key */).await?,
    }

    println!("[done] {} exiting 0", me);
    Ok(())
}
```

**Test A — `crates/famp/tests/cross_machine_happy_path.rs` (CONF-04 CI gate):**

```rust
//! CONF-04: spawn cross_machine_two_agents twice as subprocesses, ephemeral ports,
//! tempdir cert exchange. Asserts both exit 0 within a 10s timeout.

#![allow(clippy::unwrap_used, unused_crate_dependencies)]

use std::{
    process::Stdio,
    time::{Duration, Instant},
};
use tempfile::tempdir;
use tokio::{
    io::{AsyncBufReadExt, BufReader},
    process::Command,
    time::timeout,
};

#[tokio::test]
async fn cross_machine_request_commit_deliver_ack() {
    let dir = tempdir().unwrap();
    let bob_pub = dir.path().join("bob.pub");
    let bob_crt = dir.path().join("bob.crt");
    let bob_key = dir.path().join("bob.key");
    let alice_pub = dir.path().join("alice.pub");
    let alice_crt = dir.path().join("alice.crt");
    let alice_key = dir.path().join("alice.key");

    // Step 1: spawn bob with --listen 127.0.0.1:0; capture LISTENING line for actual port.
    let mut bob = Command::new(env!("CARGO_BIN_EXE_cross_machine_two_agents"))
        .args([
            "--role", "bob",
            "--listen", "127.0.0.1:0",
            "--out-pubkey", bob_pub.to_str().unwrap(),
            "--out-cert", bob_crt.to_str().unwrap(),
            "--out-key", bob_key.to_str().unwrap(),
        ])
        .stdout(Stdio::piped())
        .stderr(Stdio::piped())
        .spawn()
        .expect("spawn bob");

    let bob_stderr = bob.stderr.take().expect("bob stderr");
    let mut bob_lines = BufReader::new(bob_stderr).lines();
    let mut bob_addr = String::new();
    let deadline = Instant::now() + Duration::from_secs(5);
    while Instant::now() < deadline {
        if let Ok(Some(line)) = timeout(Duration::from_millis(500), bob_lines.next_line()).await.unwrap_or(Ok(None)).map(Some).unwrap_or(Ok(None)) {
            if let Some(rest) = line.strip_prefix("LISTENING https://") {
                bob_addr = rest.to_string();
                break;
            }
        }
    }
    assert!(!bob_addr.is_empty(), "bob did not print LISTENING");

    // Step 2: wait for bob.pub + bob.crt to exist
    let deadline = Instant::now() + Duration::from_secs(2);
    while Instant::now() < deadline && !(bob_pub.exists() && bob_crt.exists()) {
        tokio::time::sleep(Duration::from_millis(50)).await;
    }
    assert!(bob_pub.exists() && bob_crt.exists());
    let bob_pub_b64 = std::fs::read_to_string(&bob_pub).unwrap().trim().to_string();

    // Step 3: spawn alice with --peer + --addr + --trust-cert
    let alice = Command::new(env!("CARGO_BIN_EXE_cross_machine_two_agents"))
        .args([
            "--role", "alice",
            "--listen", "127.0.0.1:0",
            "--peer", &format!("agent:local/bob={}", bob_pub_b64),
            "--addr", &format!("agent:local/bob=https://{}", bob_addr),
            "--trust-cert", bob_crt.to_str().unwrap(),
            "--out-pubkey", alice_pub.to_str().unwrap(),
            "--out-cert", alice_crt.to_str().unwrap(),
            "--out-key", alice_key.to_str().unwrap(),
        ])
        .spawn()
        .expect("spawn alice");

    // Step 4: assert both exit 0 within 10s
    let alice_status = timeout(Duration::from_secs(10), alice.wait_with_output()).await
        .expect("alice timeout").expect("alice wait");
    assert!(alice_status.status.success(), "alice exit {:?}", alice_status.status);

    let bob_status = timeout(Duration::from_secs(5), bob.wait_with_output()).await
        .expect("bob timeout").expect("bob wait");
    assert!(bob_status.status.success(), "bob exit {:?}", bob_status.status);
}
```
