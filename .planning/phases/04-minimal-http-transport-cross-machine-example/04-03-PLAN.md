//! TLS helpers — rustls 0.23 only. NEVER OpenSSL, NEVER native-tls (D-B8, D-F4).
//!
//! D-B5: server loads PEM cert + private key from disk; client adds explicit
//! --trust-cert PEM as an extra RootCertStore anchor.
//!
//! Per RESEARCH.md Pitfall 4, Phase 4 takes the simpler path: the client config
//! contains ONLY the explicit --trust-cert (no platform verifier integration).
//! This is acceptable because the example only ever connects to the explicitly-
//! trusted self-signed peer. Production federation will revisit in v0.8.

use std::{fs::File, io::BufReader, path::Path};

use rustls::{
    pki_types::{CertificateDer, PrivateKeyDer},
    ClientConfig, RootCertStore, ServerConfig,
};
use rustls_pemfile::{certs, pkcs8_private_keys};

#[derive(Debug, thiserror::Error)]
pub enum TlsError {
    #[error("io error reading PEM: {0}")]
    Io(#[from] std::io::Error),
    #[error("no private key in PEM file")]
    NoPrivateKey,
    #[error("rustls error: {0}")]
    Rustls(#[from] rustls::Error),
    #[error("invalid certificate")]
    InvalidCert,
}

pub fn load_pem_cert(path: &Path) -> Result<Vec<CertificateDer<'static>>, TlsError> {
    let mut rd = BufReader::new(File::open(path)?);
    let out: Result<Vec<_>, _> = certs(&mut rd).collect();
    Ok(out?)
}

pub fn load_pem_key(path: &Path) -> Result<PrivateKeyDer<'static>, TlsError> {
    let mut rd = BufReader::new(File::open(path)?);
    let mut keys: Vec<_> = pkcs8_private_keys(&mut rd).collect::<Result<_, _>>()?;
    if keys.is_empty() {
        return Err(TlsError::NoPrivateKey);
    }
    Ok(PrivateKeyDer::Pkcs8(keys.remove(0)))
}

pub fn build_server_config(
    cert: Vec<CertificateDer<'static>>,
    key: PrivateKeyDer<'static>,
) -> Result<ServerConfig, TlsError> {
    Ok(ServerConfig::builder()
        .with_no_client_auth()
        .with_single_cert(cert, key)?)
}

/// Build a client config that trusts ONLY the explicit `--trust-cert` peer
/// (Pitfall 4 simpler-path resolution).
pub fn build_client_config_explicit_trust(
    trust_cert_path: &Path,
) -> Result<ClientConfig, TlsError> {
    let mut roots = RootCertStore::empty();
    let certs = load_pem_cert(trust_cert_path)?;
    for c in certs {
        roots.add(c).map_err(|_| TlsError::InvalidCert)?;
    }
    Ok(ClientConfig::builder()
        .with_root_certificates(roots)
        .with_no_client_auth())
}

#[cfg(test)]
mod tests {
    use super::*;
    // Round-trip test: feed a known-good fixture once Plan 04-04 generates one.
    // For Plan 04-03, just confirm the builders compile and TlsError is exported.
    #[test]
    fn tls_error_compiles() {
        let _ = TlsError::NoPrivateKey;
    }
}
```

//! `HttpTransport` — `famp_transport::Transport` impl over axum + reqwest + rustls.
//!
//! D-A2 / D-A5 / D-A6: addr_map (client) + inboxes (server) + shared reqwest::Client.
//! D-F5: native AFIT, NO async_trait macro.
//! D-C8: errors via HttpTransportError.

use std::{collections::HashMap, sync::Arc};

use famp_core::Principal;
use famp_transport::{Transport, TransportMessage};
use tokio::{
    sync::{mpsc, Mutex},
    task::JoinHandle,
};
use url::Url;

use crate::{
    error::HttpTransportError,
    server::{InboxRegistry, ServerState},
};

const INBOX_CHANNEL_CAPACITY: usize = 64;

pub struct HttpTransport {
    addr_map: Arc<Mutex<HashMap<Principal, Url>>>,
    inboxes: Arc<InboxRegistry>,
    receivers: Mutex<HashMap<Principal, mpsc::Receiver<TransportMessage>>>,
    client: reqwest::Client,
    server_handle: Mutex<Option<JoinHandle<()>>>,
}

impl HttpTransport {
    /// Construct a client-only transport with a rustls-backed reqwest client that
    /// trusts ONLY the cert at `trust_cert_path` (Pitfall 4 simpler-path resolution).
    pub fn client_only(trust_cert_path: &std::path::Path) -> Result<Self, HttpTransportError> {
        let tls = crate::tls::build_client_config_explicit_trust(trust_cert_path)
            .map_err(|_| HttpTransportError::ServerStatus { code: 0, body: "tls config failed".into() })?;
        let client = reqwest::Client::builder()
            .use_preconfigured_tls(tls)
            .timeout(std::time::Duration::from_secs(10))
            .http1_only()
            .build()
            .map_err(HttpTransportError::ReqwestFailed)?;
        Ok(Self {
            addr_map: Arc::new(Mutex::new(HashMap::new())),
            inboxes: Arc::new(Mutex::new(HashMap::new())),
            receivers: Mutex::new(HashMap::new()),
            client,
            server_handle: Mutex::new(None),
        })
    }

    pub async fn add_peer(&self, principal: Principal, url: Url) {
        self.addr_map.lock().await.insert(principal, url);
    }

    /// Register a principal as receivable on this transport. Mirrors
    /// `MemoryTransport::register` from Phase 3 D-A5.
    pub async fn register(&self, principal: Principal) {
        let (tx, rx) = mpsc::channel(INBOX_CHANNEL_CAPACITY);
        self.inboxes.lock().await.insert(principal.clone(), tx);
        self.receivers.lock().await.insert(principal, rx);
    }

    pub fn inboxes(&self) -> Arc<InboxRegistry> {
        self.inboxes.clone()
    }

    /// Take ownership of a previously-spawned server task so Drop can abort it.
    pub async fn attach_server(&self, handle: JoinHandle<()>) {
        *self.server_handle.lock().await = Some(handle);
    }
}

impl Transport for HttpTransport {
    type Error = HttpTransportError;

    fn send(
        &self,
        msg: TransportMessage,
    ) -> impl std::future::Future<Output = Result<(), Self::Error>> + Send {
        let client = self.client.clone();
        let addr_map = self.addr_map.clone();
        async move {
            let url = {
                let map = addr_map.lock().await;
                map.get(&msg.recipient).cloned().ok_or_else(|| {
                    HttpTransportError::UnknownRecipient { principal: msg.recipient.clone() }
                })?
            };
            let inbox_url = url
                .join(&format!("famp/v0.5.1/inbox/{}", msg.recipient))
                .map_err(HttpTransportError::InvalidUrl)?;
            let resp = client
                .post(inbox_url)
                .header("content-type", "application/famp+json")
                .body(msg.bytes)
                .send()
                .await
                .map_err(HttpTransportError::ReqwestFailed)?;
            let status = resp.status();
            if status == reqwest::StatusCode::ACCEPTED {
                Ok(())
            } else {
                let body = resp.text().await.unwrap_or_default();
                Err(HttpTransportError::ServerStatus { code: status.as_u16(), body })
            }
        }
    }

    fn recv(
        &self,
        principal: &Principal,
    ) -> impl std::future::Future<Output = Result<TransportMessage, Self::Error>> + Send {
        let principal = principal.clone();
        let receivers = &self.receivers;
        async move {
            let mut guard = receivers.lock().await;
            let rx = guard.get_mut(&principal).ok_or_else(|| {
                HttpTransportError::InboxClosed { principal: principal.clone() }
            })?;
            rx.recv()
                .await
                .ok_or(HttpTransportError::InboxClosed { principal })
        }
    }
}

impl Drop for HttpTransport {
    fn drop(&mut self) {
        // Best-effort abort of the server task. Graceful shutdown is deferred (Pitfall 5).
        if let Ok(mut guard) = self.server_handle.try_lock() {
            if let Some(h) = guard.take() {
                h.abort();
            }
        }
    }
}
```
