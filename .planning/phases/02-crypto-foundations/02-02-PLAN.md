       /// Returns `DOMAIN_PREFIX || canonical_json_bytes`.

       // RFC 8032 §7.1 tests exercise the Ed25519 PRIMITIVE (no FAMP prefix).
       // They use ed25519-dalek directly — the FAMP layer (DOMAIN_PREFIX) is NOT
       // applied here because RFC 8032 test vectors sign the raw message.
       use ed25519_dalek::{Signer, SigningKey, VerifyingKey, Signature};

       #[derive(serde::Deserialize)]
       struct Vector {
           name: String,
           secret_key_hex: String,
           public_key_hex: String,
           message_hex: String,
           signature_hex: String,
       }

       #[test]
       fn all_rfc8032_vectors_byte_exact() {
           let raw = include_str!("vectors/rfc8032/test-vectors.json");
           let vectors: Vec<Vector> = serde_json::from_str(raw).expect("parse vectors");
           assert_eq!(vectors.len(), 5, "RFC 8032 §7.1 has exactly 5 vectors");

           for v in &vectors {
               let sk_bytes: [u8; 32] = hex::decode(&v.secret_key_hex).unwrap().try_into().unwrap();
               let pk_bytes: [u8; 32] = hex::decode(&v.public_key_hex).unwrap().try_into().unwrap();
               let msg = hex::decode(&v.message_hex).unwrap();
               let expected_sig_bytes: [u8; 64] = hex::decode(&v.signature_hex).unwrap().try_into().unwrap();

               let sk = SigningKey::from_bytes(&sk_bytes);
               // Public key consistency
               assert_eq!(sk.verifying_key().as_bytes(), &pk_bytes, "{}: derived pk mismatch", v.name);

               // Sign → byte-exact match
               let sig = sk.sign(&msg);
               assert_eq!(sig.to_bytes(), expected_sig_bytes, "{}: signature mismatch", v.name);

               // Verify with verify_strict
               let vk = VerifyingKey::from_bytes(&pk_bytes).unwrap();
               let expected_sig = Signature::from_bytes(&expected_sig_bytes);
               vk.verify_strict(&msg, &expected_sig)
                   .unwrap_or_else(|_| panic!("{}: verify_strict failed", v.name));
           }
       }
