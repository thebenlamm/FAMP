    ```rust
    #[derive(clap::Args, Debug)]
    pub struct JoinArgs {
        /// Channel name (with or without leading #).
        pub channel: String,
        /// Override identity (D-01).
        #[arg(long = "as")]
        pub send_as: Option<String>,
    }

    pub struct JoinOutcome {
        pub channel: String,
        pub members: Vec<String>,
        pub drained: u64,
    }

    pub async fn run_at_structured(sock: &Path, args: JoinArgs) -> Result<JoinOutcome, CliError> {
        let identity = crate::cli::identity::resolve_identity(args.send_as.as_deref())?;
        let channel = crate::cli::util::normalize_channel(&args.channel)?;
        let mut bus = crate::bus_client::BusClient::connect(sock).await
            .map_err(|_| CliError::BrokerUnreachable)?;
        let reply = bus.send_recv(BusMessage::Join { channel: channel.clone() }).await
            .map_err(|e| CliError::BusClient { source: format!("{e:?}") })?;
        match reply {
            BusReply::JoinOk { members, drained } => Ok(JoinOutcome { channel, members, drained }),
            BusReply::Err { kind: BusErrorKind::NotRegistered, .. } => Err(CliError::NotRegisteredHint { name: identity }),
            BusReply::Err { kind, message } => Err(CliError::BusError { kind, message }),
            other => Err(CliError::Io { /* unexpected */ }),
        }
    }

    pub async fn run(args: JoinArgs) -> Result<(), CliError> {
        let outcome = run_at_structured(&crate::bus_client::resolve_sock_path(), args).await?;
        println!("{}", serde_json::json!({
            "channel": outcome.channel,
            "members": outcome.members,
            "drained": outcome.drained,
        }));
        Ok(())
    }
    ```

    ```rust
    #[derive(clap::Args, Debug)]
    pub struct SessionsArgs {
        /// Filter to caller's resolved identity only.
        #[arg(long)]
        pub me: bool,
    }

    pub async fn run_at(sock: &Path, args: SessionsArgs, mut out: impl std::io::Write) -> Result<(), CliError> {
        let mut bus = crate::bus_client::BusClient::connect(sock).await
            .map_err(|_| CliError::BrokerUnreachable)?;
        let reply = bus.send_recv(BusMessage::Sessions {}).await
            .map_err(|e| CliError::BusClient { source: format!("{e:?}") })?;
        match reply {
            BusReply::SessionsOk { rows } => {
                let filter: Option<String> = if args.me {
                    Some(crate::cli::identity::resolve_identity(None)?)
                } else { None };
                for row in &rows {
                    if filter.as_deref().map_or(true, |name| row.name == name) {
                        writeln!(out, "{}", serde_json::to_string(row)?)?;
                    }
                }
                Ok(())
            }
            BusReply::Err { kind, message } => Err(CliError::BusError { kind, message }),
            other => Err(CliError::Io { /* unexpected */ }),
        }
    }

    pub async fn run(args: SessionsArgs) -> Result<(), CliError> {
        run_at(&crate::bus_client::resolve_sock_path(), args, std::io::stdout().lock()).await
    }
    ```
    (Note: `sessions` is read-only — RESEARCH §6 states "no `--as`, use `--me` instead". `--me` resolves through D-01 normally; if D-01 errors (no identity bound), the `--me` flag returns that error.)

    ```rust
    #[derive(clap::Args, Debug)]
    pub struct WhoamiArgs {
        /// Override identity (D-01).
        #[arg(long = "as")]
        pub send_as: Option<String>,
    }

    pub struct WhoamiOutcome {
        pub active: Option<String>,
        pub joined: Vec<String>,
    }

    pub async fn run_at_structured(sock: &Path, args: WhoamiArgs) -> Result<WhoamiOutcome, CliError> {
        // For whoami, --as overrides the local resolution but the broker is the source of truth.
        // Implementation: send Whoami; the broker returns whatever it has for the connecting client's PID.
        // The --as flag resolves identity for hard-error message purposes only.
        let mut bus = crate::bus_client::BusClient::connect(sock).await
            .map_err(|_| CliError::BrokerUnreachable)?;
        let reply = bus.send_recv(BusMessage::Whoami {}).await
            .map_err(|e| CliError::BusClient { source: format!("{e:?}") })?;
        match reply {
            BusReply::WhoamiOk { active, joined } => Ok(WhoamiOutcome { active, joined }),
            BusReply::Err { kind, message } => Err(CliError::BusError { kind, message }),
            other => Err(CliError::Io { /* unexpected */ }),
        }
    }

    pub async fn run(args: WhoamiArgs) -> Result<(), CliError> {
        let out = run_at_structured(&crate::bus_client::resolve_sock_path(), args).await?;
        println!("{}", serde_json::json!({"active": out.active, "joined": out.joined}));
        Ok(())
    }
    ```
