    use std::process::{Command, Stdio, Child};
    use std::time::Duration;
    use std::path::Path;
    use assert_cmd::cargo::CommandCargoExt;

    struct Bus {
        _tmp: tempfile::TempDir,
        sock: std::path::PathBuf,
    }

    impl Bus {
        fn new() -> Self {
            let tmp = tempfile::TempDir::new().unwrap();
            let sock = tmp.path().join("bus.sock");
            Self { _tmp: tmp, sock }
        }
        fn env_key_val(&self) -> (&str, &Path) { ("FAMP_BUS_SOCKET", self.sock.as_path()) }
        fn famp_cmd(&self, args: &[&str]) -> std::process::Output {
            Command::cargo_bin("famp").unwrap()
                .env(self.env_key_val().0, self.env_key_val().1)
                .args(args)
                .output().unwrap()
        }
        fn famp_spawn(&self, args: &[&str]) -> Child {
            Command::cargo_bin("famp").unwrap()
                .env(self.env_key_val().0, self.env_key_val().1)
                .args(args)
                .stdin(Stdio::null()).stdout(Stdio::null()).stderr(Stdio::null())
                .spawn().unwrap()
        }
    }

    #[test]
    fn test_register_blocks() {
        // CLI-01 — already partially covered by plan 02-03; reaffirm here.
        let bus = Bus::new();
        let mut alice = bus.famp_spawn(&["register", "alice", "--no-reconnect"]);
        std::thread::sleep(Duration::from_millis(500));
        // alice should still be running

        alice.kill().ok();
        let _ = alice.wait();
    }
