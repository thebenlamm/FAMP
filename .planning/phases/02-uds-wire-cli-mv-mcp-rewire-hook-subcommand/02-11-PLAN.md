    #[tokio::test(start_paused = true)]
    async fn test_broker_idle_exit() {
        let tmp = tempfile::TempDir::new().unwrap();
        let sock = tmp.path().join("bus.sock");
        let listener = tokio::net::UnixListener::bind(&sock).unwrap();

        let sock_clone = sock.clone();
        let broker = tokio::spawn(async move {
            // shutdown_signal that never fires — only idle-exit triggers shutdown
            let never_shutdown: std::pin::Pin<Box<dyn std::future::Future<Output = ()> + Send>> =
                Box::pin(async { std::future::pending::<()>().await });
            let _ = famp::cli::broker::run_on_listener(&sock_clone, listener, never_shutdown).await;
        });

        // Connect one client and disconnect — count goes 1→0, idle timer arms.
        {
            let _stream = tokio::net::UnixStream::connect(&sock).await.unwrap();
            // drop closes; broker_rx receives Disconnect → idle = Some(Sleep(300s))
        }
        tokio::task::yield_now().await;

        // Fast-forward past 5 minutes
        tokio::time::advance(std::time::Duration::from_secs(301)).await;
        tokio::task::yield_now().await;

        // The broker task should have completed (clean shutdown via idle-exit arm)
        let join_result = tokio::time::timeout(
            std::time::Duration::from_millis(500),
            broker
        ).await;
        assert!(join_result.is_ok(), "broker must exit within 500ms after idle timer fires");
        assert!(!sock.exists(), "broker must unlink socket on idle exit");
    }
    ```

        let mut s = tokio::net::UnixStream::connect(&sock).await.unwrap();
        send_msg(&mut s, &famp_bus::BusMessage::Hello { bus_proto: 1, client: "test".into() }).await;
        let _ = recv_reply(&mut s).await;
        send_msg(&mut s, &famp_bus::BusMessage::Register { name: "real".into(), pid: std::process::id() }).await;

    #![cfg(unix)]
    #![allow(unused_crate_dependencies)]
    #![allow(clippy::unwrap_used, clippy::expect_used)]

    use std::time::Duration;
    use std::process::{Command, Stdio};
    use assert_cmd::cargo::CommandCargoExt;

    #[test]
    fn test_kill9_recovery() {
        let tmp = tempfile::TempDir::new().unwrap();
        let sock = tmp.path().join("bus.sock");
        let env_key = "FAMP_BUS_SOCKET";
        let env_val = sock.clone();

        // 1. Start alice and bob register processes
        let mut alice = Command::cargo_bin("famp").unwrap()
            .env(env_key, &env_val)
            .args(["register", "alice"])
            .stdin(Stdio::null()).stdout(Stdio::null()).stderr(Stdio::null())
            .spawn().unwrap();
        let mut bob = Command::cargo_bin("famp").unwrap()
            .env(env_key, &env_val)
            .args(["register", "bob"])
            .stdin(Stdio::null()).stdout(Stdio::null()).stderr(Stdio::null())
            .spawn().unwrap();
        std::thread::sleep(Duration::from_secs(1));

        // 2. alice sends to bob — using --as so we don't depend on FAMP_LOCAL_IDENTITY
        let send = Command::cargo_bin("famp").unwrap()
            .env(env_key, &env_val)
            .args(["send", "--as", "alice", "--to", "bob", "--new-task", "hello"])
            .output().unwrap();
        assert!(send.status.success(), "send must succeed: {}", String::from_utf8_lossy(&send.stderr));

        // 3. Find the broker pid via pgrep
        let pgrep = Command::new("pgrep")
            .args(["-f", &format!("famp broker --socket {}", sock.display())])
            .output().unwrap();
        let pid_str = String::from_utf8_lossy(&pgrep.stdout);
        let broker_pid: i32 = pid_str.lines().next()
            .and_then(|s| s.trim().parse().ok())
            .expect("broker pid found via pgrep");

        // 4. SIGKILL the broker via nix
        nix::sys::signal::kill(nix::unistd::Pid::from_raw(broker_pid), nix::sys::signal::Signal::SIGKILL).unwrap();

        // 5. Wait for the alice/bob register processes to detect the disconnect, run backoff,
        // and respawn the broker via spawn_broker_if_absent. With cap=30s starting at 1s the
        // first reconnect attempt is ~1s after the kill.
        std::thread::sleep(Duration::from_secs(3));

        // 6. Verify the message persisted: list bob's inbox
        let inbox = Command::cargo_bin("famp").unwrap()
            .env(env_key, &env_val)
            .args(["inbox", "list", "--as", "bob"])
            .output().unwrap();
        let stdout = String::from_utf8_lossy(&inbox.stdout);
        assert!(stdout.contains("hello"),
            "bob's inbox must contain 'hello' after broker kill+recovery; got: {stdout}");

        alice.kill().ok();
        bob.kill().ok();
        let _ = alice.wait();
        let _ = bob.wait();
    }
    ```
