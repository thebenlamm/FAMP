    ```rust
    use famp_bus::{BusMessage, BusReply, BusErrorKind};
    use serde_json::Value;
    use crate::cli::mcp::session;

    pub async fn call(input: &Value) -> Result<Value, BusErrorKind> {
        let name = input["name"].as_str()
            .ok_or(BusErrorKind::EnvelopeInvalid)?.to_string();
        validate_identity_name(&name)?;
        session::ensure_bus().await?;
        let mut guard = session::state().lock().await;
        let bus = guard.bus.as_mut().expect("ensure_bus succeeded");
        let pid = std::process::id();
        let reply = bus.send_recv(BusMessage::Register { name: name.clone(), pid }).await
            .map_err(|_| BusErrorKind::BrokerUnreachable)?;
        match reply {
            BusReply::RegisterOk { drained, joined } => {
                guard.active_identity = Some(name.clone());
                Ok(serde_json::json!({
                    "active": name,
                    "drained": drained.len(),
                    "peers": joined,
                }))
            }
            BusReply::Err { kind, .. } => Err(kind),
            _ => Err(BusErrorKind::Internal),
        }
    }

    fn validate_identity_name(name: &str) -> Result<(), BusErrorKind> {
        // Mirror the bash regex from scripts/famp-local cmd_register: ^[A-Za-z0-9._-]+$
        if name.is_empty() || !name.chars().all(|c| c.is_ascii_alphanumeric() || c == '.' || c == '_' || c == '-') {
            return Err(BusErrorKind::EnvelopeInvalid);
        }
        Ok(())
    }
    ```

The exhaustive-match-no-wildcard rule in CLAUDE.md is for `BusErrorKind`, `MessageClass`, `TerminalStatus` — `BusReply` may use `_` for forward compat.
