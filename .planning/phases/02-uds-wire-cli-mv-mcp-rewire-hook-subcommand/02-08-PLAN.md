    ```rust
    pub struct SessionState {
        pub bus: Option<BusClient>,
        pub active_identity: Option<String>,
    }

    pub fn state() -> &'static Mutex<SessionState> {
        static S: OnceLock<Mutex<SessionState>> = OnceLock::new();
        S.get_or_init(|| Mutex::new(SessionState { bus: None, active_identity: None }))
    }

    /// Open the BusClient if not already connected. Idempotent. Returns Ok(()) on success.
    pub async fn ensure_bus() -> Result<(), BusErrorKind> {
        let mut guard = state().lock().await;
        if guard.bus.is_none() {
            let sock = crate::bus_client::resolve_sock_path();
            let client = BusClient::connect(&sock).await.map_err(|_| BusErrorKind::BrokerUnreachable)?;
            guard.bus = Some(client);
        }
        Ok(())
    }

    /// Read-only accessor for the active identity (None until famp_register succeeds).
    pub async fn active_identity() -> Option<String> {
        state().lock().await.active_identity.clone()
    }

    /// Sets the active identity (called by famp_register after RegisterOk).
    pub async fn set_active_identity(name: String) {
        state().lock().await.active_identity = Some(name);
    }

    /// Clears session state (used by tests).
    #[cfg(test)]
    pub async fn clear() {
        let mut guard = state().lock().await;
        guard.bus = None;
        guard.active_identity = None;
    }
    ```
