    ```rust
    pub async fn run_at_structured(sock: &Path, args: AwaitArgs) -> Result<AwaitOutcome, CliError> {
        let identity = crate::cli::identity::resolve_identity(args.act_as.as_deref())?;
        let timeout_ms: u64 = std::time::Duration::from(args.timeout).as_millis().try_into().unwrap_or(u64::MAX);
        // Connect with bind_as proxy per D-10. The broker validates the named identity is held by
        // a live `famp register` at Hello time and re-verifies on each op.
        let mut bus = crate::bus_client::BusClient::connect(sock, Some(identity.clone())).await
            .map_err(|e| match e {
                BusClientError::HelloFailed { kind: BusErrorKind::NotRegistered, .. } => {
                    CliError::NotRegisteredHint { name: identity.clone() }
                }
                _ => CliError::BrokerUnreachable,
            })?;
        let reply = bus.send_recv(BusMessage::Await { timeout_ms, task: args.task }).await
            .map_err(|e| CliError::BusClient { source: format!("{e:?}") })?;
        match reply {
            BusReply::AwaitOk { envelope } => Ok(AwaitOutcome { envelope: Some(envelope), timed_out: false }),
            BusReply::AwaitTimeout {} => Ok(AwaitOutcome { envelope: None, timed_out: true }),
            BusReply::Err { kind: BusErrorKind::NotRegistered, .. } => {
                // Per-op liveness re-check failed (holder died between Hello and Await)
                Err(CliError::NotRegisteredHint { name: identity })
            }
            BusReply::Err { kind, message } => Err(CliError::BusError { kind, message }),
            other => Err(CliError::Io { /* unexpected */ }),
        }
    }

    pub async fn run_at(sock: &Path, args: AwaitArgs, mut out: impl std::io::Write) -> Result<(), CliError> {
        let outcome = run_at_structured(sock, args).await?;
        if outcome.timed_out {
            writeln!(out, "{}", serde_json::json!({"timeout": true}))?;
        } else if let Some(env) = outcome.envelope {
            writeln!(out, "{}", serde_json::to_string(&env)?)?;
        }
        Ok(())
    }

    pub async fn run(args: AwaitArgs) -> Result<(), CliError> {
        run_at(&crate::bus_client::resolve_sock_path(), args, std::io::stdout().lock()).await
    }
    ```
