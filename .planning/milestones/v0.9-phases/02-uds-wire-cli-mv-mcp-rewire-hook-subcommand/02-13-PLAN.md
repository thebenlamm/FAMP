    #![cfg(unix)]
    #![allow(unused_crate_dependencies)]
    #![allow(clippy::unwrap_used, clippy::expect_used)]

    use std::io::{BufRead, BufReader, Read, Write};
    use std::process::{Child, ChildStdin, ChildStdout, Command, Stdio};
    use std::path::Path;
    use std::time::Duration;
    use assert_cmd::cargo::CommandCargoExt;
    use serde_json::{json, Value};

    struct McpHarness {
        child: Child,
        stdin: ChildStdin,
        stdout: BufReader<ChildStdout>,
        next_id: u64,
    }

    impl McpHarness {
        fn spawn(sock: &Path, label: &str) -> Self {
            let mut child = Command::cargo_bin("famp").unwrap()
                .args(["mcp"])
                .env("FAMP_BUS_SOCKET", sock)
                .env_remove("FAMP_HOME")
                .env_remove("FAMP_LOCAL_ROOT")
                .stdin(Stdio::piped()).stdout(Stdio::piped()).stderr(Stdio::null())
                .spawn().unwrap_or_else(|e| panic!("{label}: spawn failed: {e}"));
            let stdin = child.stdin.take().unwrap();
            let stdout = BufReader::new(child.stdout.take().unwrap());
            Self { child, stdin, stdout, next_id: 1 }
        }

        fn send(&mut self, method: &str, params: Value) {
            let id = self.next_id; self.next_id += 1;
            let msg = json!({"jsonrpc":"2.0","id":id,"method":method,"params":params}).to_string();
            let frame = format!("Content-Length: {}\r\n\r\n{}", msg.len(), msg);
            self.stdin.write_all(frame.as_bytes()).unwrap();
            self.stdin.flush().unwrap();
        }

        fn recv(&mut self) -> Value {
            // Read headers up to blank line
            let mut content_length: Option<usize> = None;
            loop {
                let mut line = String::new();
                self.stdout.read_line(&mut line).unwrap();
                let trimmed = line.trim_end();
                if trimmed.is_empty() { break; }
                if let Some(rest) = trimmed.strip_prefix("Content-Length:") {
                    content_length = Some(rest.trim().parse().unwrap());
                }
            }
            let len = content_length.expect("Content-Length header");
            let mut buf = vec![0u8; len];
            self.stdout.get_mut().read_exact(&mut buf).unwrap();
            serde_json::from_slice(&buf).unwrap()
        }

        fn call(&mut self, tool: &str, args: Value) -> Value {
            self.send("tools/call", json!({"name": tool, "arguments": args}));
            let reply = self.recv();
            reply
        }

        fn shutdown(mut self) {
            self.child.kill().ok();
            let _ = self.child.wait();
        }
    }

    #[test]
    fn test_mcp_bus_e2e() {
        let tmp = tempfile::TempDir::new().unwrap();
        let sock = tmp.path().join("test-bus.sock");

        // Pre-spawn a broker via a third subprocess (the broker subcommand)
        // OR rely on the first MCP process to spawn it via spawn_broker_if_absent.
        // Use the first-MCP-spawns approach for true bus-spawn-on-demand testing.

        let mut alice = McpHarness::spawn(&sock, "alice");
        let mut bob = McpHarness::spawn(&sock, "bob");

        // Give the broker time to spawn (alice's first famp_register call triggers spawn)
        // We'll measure latency by sending the register and waiting for reply.

        // 1. alice registers
        let reg_alice = alice.call("famp_register", json!({"name": "alice"}));
        assert!(reg_alice.get("error").is_none(),
            "alice register error: {reg_alice}");
        let result = &reg_alice["result"];
        assert_eq!(result["active"], "alice");

        // 2. bob registers
        let reg_bob = bob.call("famp_register", json!({"name": "bob"}));
        assert!(reg_bob.get("error").is_none(),
            "bob register error: {reg_bob}");
        let result = &reg_bob["result"];
        assert_eq!(result["active"], "bob");

        // 3. alice sends to bob
        let send = alice.call("famp_send", json!({
            "to": {"kind": "agent", "name": "bob"},
            "new_task": "hello from alice"
        }));
        assert!(send.get("error").is_none(),
            "send error: {send}");
        let task_id = send["result"]["task_id"].as_str().expect("task_id");
        assert!(!task_id.is_empty());

        // 4. bob awaits — should unblock with the message
        let awaited = bob.call("famp_await", json!({"timeout_ms": 5000}));
        assert!(awaited.get("error").is_none(),
            "await error: {awaited}");
        let envelope = &awaited["result"]["envelope"];
        assert!(!envelope.is_null(), "expected envelope, got: {awaited}");
        // Check from + body. The exact JSON path depends on v0.5.2 envelope schema.
        // The "from" field is at the top level of the envelope.
        assert_eq!(envelope["from"], "alice", "envelope: {envelope}");
        // body field contains the task body
        let body = &envelope["body"];
        assert!(body.to_string().contains("hello from alice"),
            "body should contain task text; got: {body}");

        alice.shutdown();
        bob.shutdown();
    }
