    ```rust
    async fn test_broker_idle_exit() {
        let tmp = tempfile::TempDir::new().unwrap();
        let sock = tmp.path().join("bus.sock");
        let listener = tokio::net::UnixListener::bind(&sock).unwrap();

        let sock_clone = sock.clone();
        let broker = tokio::spawn(async move {
            // shutdown_signal that never fires — only idle-exit triggers shutdown
            let never_shutdown: std::pin::Pin<Box<dyn std::future::Future<Output = ()> + Send>> =
                Box::pin(async { std::future::pending::<()>().await });
            let _ = famp::cli::broker::run_on_listener(&sock_clone, listener, never_shutdown).await;
        });

        // Connect one client and disconnect — count goes 1→0, idle timer arms.
        {
            let _stream = tokio::net::UnixStream::connect(&sock).await.unwrap();
            // drop closes; broker_rx receives Disconnect → idle = Some(Sleep(300s))
        }
        tokio::task::yield_now().await;

        // Fast-forward past 5 minutes
        tokio::time::advance(std::time::Duration::from_secs(301)).await;
        tokio::task::yield_now().await;

        // The broker task should have completed (clean shutdown via idle-exit arm)
        let join_result = tokio::time::timeout(
            std::time::Duration::from_millis(500),
            broker
        ).await;
        assert!(join_result.is_ok(), "broker must exit within 500ms after idle timer fires");
        assert!(!sock.exists(), "broker must unlink socket on idle exit");
    }
    ```

    ```rust
        // 1. Start alice and bob register processes

            .env(env_key, &env_val)
            .args(["register", "bob"])
    ```

    ```rust
    #![cfg(unix)]
    #![allow(unused_crate_dependencies)]
    #![allow(clippy::unwrap_used, clippy::expect_used)]

    use std::time::Duration;
    use std::process::{Command, Stdio};
    use assert_cmd::cargo::CommandCargoExt;

    /// D-10 invariant: a proxy connection (--as alice from a one-shot CLI)
    /// joining a channel adds alice to the channel via the canonical holder's
    /// joined-set. The proxy disconnect (one-shot CLI exits) MUST NOT remove
    /// alice from the channel.
    #[test]
    fn test_proxy_join_persists_after_disconnect() {
        let tmp = tempfile::TempDir::new().unwrap();
        let sock = tmp.path().join("bus.sock");
        let env_key = "FAMP_BUS_SOCKET";

        // 1. Start alice as canonical holder
        let mut alice = Command::cargo_bin("famp").unwrap()
            .env(env_key, &sock)
            .args(["register", "alice"])
            .stdin(Stdio::null()).stdout(Stdio::null()).stderr(Stdio::null())
            .spawn().unwrap();
        std::thread::sleep(Duration::from_secs(1));

        // 2. One-shot proxy: famp join --as alice #planning (this is a SEPARATE process
        //    from the long-lived `famp register alice` — it connects with bind_as: Some("alice")
        //    per D-10 and then EXITS. The broker MUST NOT auto-leave alice from #planning.)
        let join = Command::cargo_bin("famp").unwrap()
            .env(env_key, &sock)
            .args(["join", "#planning", "--as", "alice"])
            .output().unwrap();
        assert!(join.status.success(), "join failed: {}", String::from_utf8_lossy(&join.stderr));

        // 3. The proxy `famp join` process has now exited. Verify alice is STILL in #planning
        //    by querying sessions from a separate one-shot process (also uses bind_as proxy).
        std::thread::sleep(Duration::from_millis(500));
        let sessions = Command::cargo_bin("famp").unwrap()
            .env(env_key, &sock)
            .args(["sessions"])
            .output().unwrap();
        assert!(sessions.status.success());
        let stdout = String::from_utf8_lossy(&sessions.stdout);
        // alice's row should include #planning in its joined list
        assert!(stdout.contains("alice"), "alice must be in sessions: {stdout}");
        assert!(stdout.contains("#planning"),
            "alice MUST still be in #planning after proxy disconnect (D-10 invariant): {stdout}");

        alice.kill().ok();
        let _ = alice.wait();
    }

    /// D-10 invariant: --as alice when no `famp register alice` process is live
    /// MUST exit non-zero with a NotRegistered hint. The broker rejects the proxy
    /// at Hello time (HelloErr{NotRegistered}); the CLI surfaces this as the user-facing
    /// hint message.
    #[test]
    fn test_proxy_inbox_unregistered_fails() {
        let tmp = tempfile::TempDir::new().unwrap();
        let sock = tmp.path().join("bus.sock");
        let env_key = "FAMP_BUS_SOCKET";

        // No `famp register alice` running. The first connect spawns a broker, but the
        // proxy Hello.bind_as = Some("alice") MUST fail because alice is not held by any
        // live registered process.
        let inbox = Command::cargo_bin("famp").unwrap()
            .env(env_key, &sock)
            .args(["inbox", "list", "--as", "alice"])
    ```
