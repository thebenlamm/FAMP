    Create or rewrite `crates/famp/src/cli/inbox/ack.rs`:

    ```rust
    #[derive(clap::Args, Debug)]
    pub struct AckArgs {
        /// Cursor offset to advance to. Required; obtain from `famp inbox list` next_offset.
        #[arg(long)]
        pub offset: u64,
        /// Override identity (D-01).
        #[arg(long = "as")]
        pub act_as: Option<String>,
    }

    pub async fn run(args: AckArgs) -> Result<(), CliError> {
        let identity = crate::cli::identity::resolve_identity(args.act_as.as_deref())?;
        let sock = crate::bus_client::resolve_sock_path();
        let bus_dir = crate::bus_client::bus_dir(&sock);
        // CRITICAL: NO BusClient::connect call here. The cursor is local — NO broker round-trip,
        // NO Hello.bind_as proxy required. Per RESEARCH §6 the client is authoritative on cursor state.
        crate::cli::broker::cursor_exec::execute_advance_cursor(bus_dir, &identity, args.offset)
            .await
            .map_err(|e| CliError::Io { path: bus_dir.to_path_buf(), source: e })?;
        println!("{}", serde_json::json!({ "acked": true, "offset": args.offset }));
        Ok(())
    }
    ```
