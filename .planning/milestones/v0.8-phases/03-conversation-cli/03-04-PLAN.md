```rust
// crates/famp-inbox/src/lock.rs
use std::path::{Path, PathBuf};

pub struct InboxLock { path: PathBuf, _file: std::fs::File }

impl InboxLock {
    /// Acquire the advisory lock at `<home>/inbox.lock`. Best-effort
    /// PID file: writes the current process id to the lock file. On
    /// next acquire, if the file exists AND the PID is still alive,
    /// returns LockHeld. If the PID is dead (process gone), the lock
    /// is considered stale and reacquired.
    ///
    /// Mode 0600 on Unix.
    pub fn acquire(home: &Path) -> Result<Self, InboxError>;

    pub fn path(&self) -> &Path;
}

impl Drop for InboxLock {
    fn drop(&mut self) {
        // Best-effort: remove the lock file. If the rm fails (e.g.
        // another process raced and re-created it), do nothing.
        let _ = std::fs::remove_file(&self.path);
    }
}
```

New `InboxError` variant:
```rust
#[error("inbox lock held by pid {pid} at {path:?}")]
LockHeld { path: PathBuf, pid: u32 },
```

**Locking semantics decision (lock here):** `famp await` acquires the
lock IMMEDIATELY on entry. If the lock is held by a live PID, returns
`CliError::Inbox(InboxError::LockHeld { .. })` and exits non-zero
WITHOUT polling. (Alternative: spin-wait until the lock is free. We
choose fail-fast because Phase 3's CLI is a single-developer surface
— concurrent awaits are user error and silent waiting would mask the
mistake.)

```rust
//! Advisory file-based inbox lock. Best-effort, RAII.
//!
//! Held by `famp await` (and any future single-consumer reader)
//! to prevent two processes from double-consuming inbox entries.
//! The lock file contains the holder's PID; stale PIDs (dead
//! processes) are reaped on next acquire.

use std::fs::OpenOptions;
use std::io::{Read as _, Write as _};
use std::path::{Path, PathBuf};

use crate::InboxError;

pub struct InboxLock {
    path: PathBuf,
    _file: std::fs::File, // hold the handle for the duration
}

impl InboxLock {
    pub fn acquire(home: &Path) -> Result<Self, InboxError> {
        let path = home.join("inbox.lock");

        // Check existing lock for liveness.
        if path.exists() {
            if let Ok(mut f) = std::fs::File::open(&path) {
                let mut s = String::new();
                if f.read_to_string(&mut s).is_ok() {
                    if let Ok(pid) = s.trim().parse::<u32>() {
                        if is_alive(pid) {
                            return Err(InboxError::LockHeld {
                                path: path.clone(), pid,
                            });
                        }
                    }
                }
            }
            // Stale or unparseable — reap.
            let _ = std::fs::remove_file(&path);
        }

        // Create exclusively. If a race lost, return LockHeld.
        let mut opts = OpenOptions::new();
        opts.write(true).create_new(true);
        #[cfg(unix)]
        {
            use std::os::unix::fs::OpenOptionsExt;
            opts.mode(0o600);
        }
        let mut file = match opts.open(&path) {
            Ok(f) => f,
            Err(e) if e.kind() == std::io::ErrorKind::AlreadyExists => {
                // Race: someone else created the file between our
                // exists() check and now. Re-read its PID.
                let pid = std::fs::read_to_string(&path)
                    .ok()
                    .and_then(|s| s.trim().parse::<u32>().ok())
                    .unwrap_or(0);
                return Err(InboxError::LockHeld { path, pid });
            }
            Err(source) => {
                return Err(InboxError::Io { path, source });
            }
        };

        let pid = std::process::id();
        file.write_all(format!("{pid}\n").as_bytes())
            .map_err(|source| InboxError::Io { path: path.clone(), source })?;
        file.sync_all()
            .map_err(|source| InboxError::Io { path: path.clone(), source })?;

        Ok(Self { path, _file: file })
    }

    pub fn path(&self) -> &Path { &self.path }
}

impl Drop for InboxLock {
    fn drop(&mut self) {
        let _ = std::fs::remove_file(&self.path);
    }
}

#[cfg(unix)]
fn is_alive(pid: u32) -> bool {
    use nix::sys::signal::kill;
    use nix::unistd::Pid;
    // signal=None → just check existence. EPERM = exists but owned
    // by another user, still alive. ESRCH = dead.
    match kill(Pid::from_raw(pid as i32), None) {
        Ok(()) => true,
        Err(nix::errno::Errno::EPERM) => true,
        _ => false,
    }
}

#[cfg(not(unix))]
fn is_alive(_pid: u32) -> bool {
    // Conservative: assume alive on non-Unix. Phase 3 doesn't target
    // Windows so this branch is unreachable in practice.
    true
}
```

```rust
mod common;
use common::conversation_harness::*;

#[tokio::test]
async fn full_long_task_conversation_completes() {
    let (alice_dir, bob_dir) = setup_two_homes().await;
    let alice = alice_dir.path();
    let bob = bob_dir.path();

    // Spawn both listeners (ephemeral ports).
    let (alice_addr, alice_handle, alice_shutdown) = spawn_listener(alice).await;
    let (bob_addr,   bob_handle,   bob_shutdown)   = spawn_listener(bob).await;

    // Cross-register peers.
    peer_add_each_other(alice, bob, alice_addr, bob_addr).await;

    // 1. Alice opens a new task to Bob.
    let task_id = run_send_new_task(alice, "bob", "hello bob").await;

    // 2. Three non-terminal delivers.
    for i in 0..3 {
        run_send_deliver(alice, "bob", &task_id, false, &format!("msg {i}")).await;
    }

    // 3. Bob's inbox should now hold 4 lines (1 request + 3 deliver).
    let bob_inbox = bob.join("inbox.jsonl");
    let lines: Vec<_> = std::fs::read_to_string(&bob_inbox).unwrap().lines().collect();
    assert_eq!(lines.len(), 4, "bob should have received 4 envelopes");

    // 4. Alice's task record stays non-terminal across the 3 sends.
    let rec = read_task_record(alice, &task_id);
    assert_eq!(rec.state, "REQUESTED");
    assert!(!rec.terminal);

    // 5. Terminal deliver.
    run_send_deliver(alice, "bob", &task_id, true, "done").await;

    // 6. Alice's record is now COMPLETED + terminal.
    let rec = read_task_record(alice, &task_id);
    assert_eq!(rec.state, "COMPLETED");
    assert!(rec.terminal);

    // 7. Subsequent send on the same task fails TaskTerminal.
    let err = std::panic::catch_unwind(|| {
        // run_send_deliver panics on Err — call the underlying API
        // directly here:
    });
    // (alternative: extend harness with `try_run_send_deliver` that
    //  returns `Result<(), CliError>` for negative tests)
    let result = try_run_send_deliver(alice, "bob", &task_id, false, "ghost").await;
```

```rust
#[tokio::test]
async fn second_await_while_first_holds_lock_returns_lockheld() {
    let (alice_dir, _bob_dir) = setup_two_homes().await;
    let alice = alice_dir.path();

    // Acquire the lock manually to simulate a held first await.
    let lock = famp_inbox::InboxLock::acquire(alice).expect("acquire");

    // Run famp await — should fail fast with LockHeld.
    let result = famp::cli::await_cmd::run_at(
        alice,
        famp::cli::await_cmd::AwaitArgs {
            timeout: "5s".to_string(),
            task: None,
        },
        &mut std::io::sink(),
    ).await;

    assert!(matches!(
        result,
        Err(famp::cli::error::CliError::Inbox(famp_inbox::InboxError::LockHeld { .. }))
    ));

    drop(lock);

    // After release, await on an empty inbox times out (no entries),
    // proving the lock no longer blocks acquisition.
    let result2 = famp::cli::await_cmd::run_at(
        alice,
        famp::cli::await_cmd::AwaitArgs {
            timeout: "100ms".to_string(),
            task: None,
        },
        &mut std::io::sink(),
    ).await;
    assert!(matches!(result2, Err(famp::cli::error::CliError::AwaitTimeout { .. })));
}
```
