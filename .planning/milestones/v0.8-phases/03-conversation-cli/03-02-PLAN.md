<objective>
Wire the outbound half of the conversation CLI:

- `famp peer add` registers a peer in `peers.toml` so `famp send --to`
  can resolve aliases.
- `famp send` covers all three modes — `--new-task`, `--task` (non-terminal),
  `--task --terminal` — using the v0.7 envelope, signing, and FSM crates
  with no spec changes. Persists task state to `~/.famp/tasks/<uuid>.toml`
  via the `famp-taskdir` primitive from Plan 03-01.
- TLS contact uses TOFU: first send to a peer records the leaf cert's
  SHA-256 into `peers.toml`; subsequent sends pin against it. Mirrors
  the v0.7 keyring TOFU pattern.

Purpose: half of Phase 3's success criteria (1, 2, 3, 5) depend on
`famp send` working correctly with persistent task records. This plan
delivers that surface.

```rust
//! `famp peer add` — register a peer alias in peers.toml.

use std::path::Path;

use crate::cli::config::{PeerEntry, Peers};
use crate::cli::error::CliError;
use crate::cli::{home, paths};

pub fn run_add(alias: String, endpoint: String, pubkey_b64: String)
    -> Result<(), CliError>
{
    let home = home::resolve_famp_home()?;
    run_add_at(&home, alias, endpoint, pubkey_b64)
}

pub fn run_add_at(
    home: &Path,
    alias: String,
    endpoint: String,
    pubkey_b64: String,
) -> Result<(), CliError> {
    // 1. Validate endpoint — must be https URL.
    let parsed = url::Url::parse(&endpoint)
        .map_err(|_| CliError::PeerEndpointInvalid { value: endpoint.clone() })?;
    if parsed.scheme() != "https" {
        return Err(CliError::PeerEndpointInvalid { value: endpoint });
    }

    // 2. Validate pubkey — base64url-unpadded → 32 bytes.
    use base64::engine::general_purpose::URL_SAFE_NO_PAD;
    use base64::Engine as _;
    let raw = URL_SAFE_NO_PAD.decode(pubkey_b64.as_bytes())
        .map_err(|_| CliError::PeerPubkeyInvalid { value: pubkey_b64.clone() })?;
    if raw.len() != 32 {
        return Err(CliError::PeerPubkeyInvalid { value: pubkey_b64 });
    }

    // 3. Load peers.toml, append, write back atomically.
    let path = paths::peers_toml_path(home);
    let bytes = std::fs::read(&path).map_err(|e| CliError::Io {
        path: path.clone(), source: e,
    })?;
    let mut peers: Peers = if bytes.is_empty() {
        Peers::default()
    } else {
        toml::from_slice(&bytes).map_err(|e| CliError::TomlParse {
            path: path.clone(), source: e,
        })?
        // (toml 0.8 prefers `from_str(std::str::from_utf8(...)?)` — adapt
        //  to whichever API the workspace already uses; check Phase 1's
        //  config loader.)
    };
    let entry = PeerEntry {
        alias: alias.clone(),
        endpoint,
        pubkey_b64,
        tls_fingerprint_sha256: None,
    };
    if peers.try_add(entry).is_err() {
        return Err(CliError::PeerDuplicate { alias });
    }
    let serialized = toml::to_string(&peers).map_err(CliError::TomlSerialize)?;

    // Atomic replace via tempfile in the same dir.
    write_atomic(&path, serialized.as_bytes())?;
    Ok(())
}

fn write_atomic(path: &Path, bytes: &[u8]) -> Result<(), CliError> {
    use std::io::Write as _;
    let parent = path.parent().ok_or_else(|| CliError::HomeHasNoParent {
        path: path.to_path_buf(),
    })?;
    let mut tmp = tempfile::NamedTempFile::new_in(parent).map_err(|e| CliError::Io {
        path: parent.to_path_buf(), source: e,
    })?;
    tmp.write_all(bytes).map_err(|e| CliError::Io {
        path: path.to_path_buf(), source: e,
    })?;
    tmp.as_file_mut().sync_all().map_err(|e| CliError::Io {
        path: path.to_path_buf(), source: e,
    })?;
    tmp.persist(path).map_err(|e| CliError::Io {
        path: path.to_path_buf(), source: e.error,
    })?;
    Ok(())
}
```

**2. Create `crates/famp/src/cli/send/fsm_glue.rs`:**

```rust
//! Bridge between the on-disk TaskRecord (string state) and the
//! in-memory `famp_fsm::TaskFsm` (typed state).

use famp_core::{MessageClass, TerminalStatus};
use famp_fsm::{TaskFsm, TaskState, TaskTransitionInput};
use famp_taskdir::TaskRecord;

use crate::cli::error::CliError;

pub fn state_to_str(s: TaskState) -> &'static str {
    match s {
        TaskState::Requested => "REQUESTED",
        TaskState::Committed => "COMMITTED",
        TaskState::Completed => "COMPLETED",
        TaskState::Failed    => "FAILED",
        TaskState::Cancelled => "CANCELLED",
    }
}

pub fn str_to_state(s: &str) -> Result<TaskState, CliError> {
    Ok(match s {
        "REQUESTED" => TaskState::Requested,
        "COMMITTED" => TaskState::Committed,
        "COMPLETED" => TaskState::Completed,
        "FAILED"    => TaskState::Failed,
        "CANCELLED" => TaskState::Cancelled,
        _ => return Err(CliError::TaskNotFound { task_id: s.to_string() }),
    })
}

pub fn is_terminal(s: TaskState) -> bool {
    matches!(s, TaskState::Completed | TaskState::Failed | TaskState::Cancelled)
}

/// Apply a terminal deliver to an in-memory FSM seeded from the record.
/// Returns the new state.
///
/// **Important:** v0.7's `TaskFsm` only allows
/// `Committed → Completed/Failed` on a deliver-with-terminal-status.
/// Phase 3 `famp send --new-task` issues a `request` (state stays
/// REQUESTED locally), and the local sender does NOT step the FSM
/// forward to COMMITTED (the responder would do that on the
/// `commit` reply, which Phase 3 does not yet round-trip). So a
/// terminal deliver from the LOCAL sender is only legal if the
/// local record is already in COMMITTED — which Phase 3 cannot
/// achieve without a commit reply.
///
/// **Phase 3 narrowing:** the local terminal-send path SEEDS the FSM
/// at COMMITTED before stepping (the responder's commit is implicit
/// for Phase 3 — Phase 4 will round-trip it for real). Document this
/// inline as a known-Phase-3-shortcut and add a TODO pointing at
/// Phase 4.
pub fn advance_terminal(record: &mut TaskRecord) -> Result<TaskState, CliError> {
    // Phase 3 shortcut: seed at Committed regardless of record.state to
    // satisfy v0.7 FSM legality. Phase 4 will replace this with a real
    // commit-reply round-trip.
    let mut fsm = TaskFsm::__with_state_for_testing(TaskState::Committed);
    let next = fsm.step(TaskTransitionInput {
        class: MessageClass::Deliver,
        terminal_status: Some(TerminalStatus::Completed),
    }).map_err(|e| CliError::Envelope(Box::new(e)))?;
    record.state = state_to_str(next).to_string();
    record.terminal = is_terminal(next);
    Ok(next)
}
```

//!

//!   into a shared cell, accepts the cert
//! - on subsequent contacts (`pinned = Some(hex)`): computes the leaf
//!   SHA-256 and rejects if it differs from the pinned value
//!
//! After a successful first send, the caller reads the cell and
//! persists the fingerprint into peers.toml.

use std::sync::{Arc, Mutex};
use rustls::client::danger::{ServerCertVerifier, ServerCertVerified, HandshakeSignatureValid};
use rustls::pki_types::{CertificateDer, ServerName, UnixTime};
use rustls::{DigitallySignedStruct, SignatureScheme};
use sha2::{Digest, Sha256};

#[derive(Debug)]
pub struct TofuVerifier {
    pinned: Option<String>,                          // hex sha256
    captured: Arc<Mutex<Option<String>>>,            // first-seen sha256
}

impl TofuVerifier {
    pub fn new(pinned: Option<String>, captured: Arc<Mutex<Option<String>>>) -> Self {
        Self { pinned, captured }
    }
}

impl ServerCertVerifier for TofuVerifier {
    fn verify_server_cert(
        &self,
        end_entity: &CertificateDer<'_>,
        _intermediates: &[CertificateDer<'_>],
        _server_name: &ServerName<'_>,
        _ocsp_response: &[u8],
        _now: UnixTime,
    ) -> Result<ServerCertVerified, rustls::Error> {
        let mut h = Sha256::new();
        h.update(end_entity.as_ref());
        let got = hex::encode(h.finalize());
        if let Some(pinned) = &self.pinned {
            if pinned != &got {
                return Err(rustls::Error::General(format!(
                    "tls fingerprint mismatch: pinned={pinned}, got={got}"
                )));
            }
        } else {
            *self.captured.lock().unwrap() = Some(got);
        }
        Ok(ServerCertVerified::assertion())
    }

    // The two signature-validation methods just return Ok — we trust
    // the leaf because we have its fingerprint pinned (or are TOFU-ing).
    // Implementations follow the rustls 0.23 pattern from
    // famp-transport-http/src/tls.rs — copy that file's structure
    // verbatim for the TLS 1.2 / 1.3 sig schemes the runtime actually
    // negotiates.
    fn verify_tls12_signature(&self, _: &[u8], _: &CertificateDer<'_>, _: &DigitallySignedStruct)
        -> Result<HandshakeSignatureValid, rustls::Error> { Ok(HandshakeSignatureValid::assertion()) }
    fn verify_tls13_signature(&self, _: &[u8], _: &CertificateDer<'_>, _: &DigitallySignedStruct)
        -> Result<HandshakeSignatureValid, rustls::Error> { Ok(HandshakeSignatureValid::assertion()) }
    fn supported_verify_schemes(&self) -> Vec<SignatureScheme> {
        // Mirror the set famp-transport-http/src/tls.rs accepts.
        vec![
            SignatureScheme::ED25519,
            SignatureScheme::ECDSA_NISTP256_SHA256,
            SignatureScheme::RSA_PSS_SHA256,
        ]
    }
}

pub struct SendOutcome {
    pub captured_fingerprint: Option<String>,        // Some(hex) if TOFU
}

pub async fn post_envelope(
    endpoint: &str,
    recipient_principal: &str,
    bytes: Vec<u8>,
    pinned: Option<String>,
) -> Result<SendOutcome, crate::cli::error::CliError> {
    use crate::cli::error::CliError;
    let captured = Arc::new(Mutex::new(None));
    let verifier = Arc::new(TofuVerifier::new(pinned, captured.clone()));
    let tls = rustls::ClientConfig::builder()
        .dangerous()
        .with_custom_certificate_verifier(verifier)
        .with_no_client_auth();
    let client = reqwest::Client::builder()
        .use_preconfigured_tls(tls)
        .build()
        .map_err(|e| CliError::SendFailed(Box::new(e)))?;
    let url = format!("{}/famp/v0.5.1/inbox/{}", endpoint.trim_end_matches('/'), recipient_principal);
    let resp = client.post(&url)
        .header("content-type", "application/json")
        .body(bytes)
        .send().await
        .map_err(|e| CliError::SendFailed(Box::new(e)))?;
    if !resp.status().is_success() {
        return Err(CliError::SendFailed(Box::new(std::io::Error::other(
            format!("HTTP {}", resp.status()),
        ))));
    }
    let captured = captured.lock().unwrap().clone();
    Ok(SendOutcome { captured_fingerprint: captured })
}

pub async fn run(args: SendArgs) -> Result<(), CliError> {
    let home = home::resolve_famp_home()?;

}

pub async fn run_at(home: &Path, args: SendArgs) -> Result<(), CliError> {
    // 1. Load identity (signing key, principal).
    let layout = init::load_identity(home)?;
    let signing_key = load_signing_key(&layout)?;       // helper, see below
    let from_principal = principal_from_layout(&layout)?;

    // 2. Load peers, find the alias.
    let peers_path = paths::peers_toml_path(home);
    let mut peers = read_peers(&peers_path)?;
    let peer = peers.find(&args.to)
        .ok_or_else(|| CliError::PeerNotFound { alias: args.to.clone() })?
        .clone();

    // 3. Open task dir.
    let tasks = TaskDir::open(paths::tasks_dir(home))?;

    // 4. Branch on mode.
    let (envelope_bytes, task_id, mode) = match (&args.new_task, &args.task, args.terminal) {
        (Some(text), None, false) => build_new_task(&signing_key, &from_principal, &peer, text)?,
        (None, Some(id), false)   => build_deliver(&signing_key, &from_principal, &peer, id, &tasks, false, args.body.as_deref())?,
        (None, Some(id), true)    => build_deliver(&signing_key, &from_principal, &peer, id, &tasks, true,  args.body.as_deref())?,
        _ => return Err(CliError::SendFailed(Box::new(std::io::Error::other(
            "exactly one of --new-task or --task must be provided",
        )))),
    };

    // 5. Pre-flight: if --task, refuse if record is already terminal.
    if let Some(id) = &args.task {
        let record = tasks.read(id)?;
        if record.terminal {
            return Err(CliError::TaskTerminal { task_id: id.clone() });
        }
    }

    // 6. Send.
    let recipient = peer_principal_from_pubkey(&peer.pubkey_b64)?;
    let outcome = client::post_envelope(
        &peer.endpoint,
        &recipient,
        envelope_bytes,
        peer.tls_fingerprint_sha256.clone(),
    ).await?;

    // 7. Persist task record + TOFU fingerprint.
    persist_post_send(&tasks, &task_id, &args.to, mode, home, &mut peers, &peers_path, &args.to, outcome.captured_fingerprint)?;

    // 8. On --new-task mode, print the uuid.
    if matches!(mode, SendMode::NewTask) {
        println!("{task_id}");
    }
    Ok(())
}
