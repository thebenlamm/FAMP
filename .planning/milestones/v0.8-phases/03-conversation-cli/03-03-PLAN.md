**1. Extend `crates/famp-inbox/src/read.rs`:**

Add `read_from` alongside `read_all`. Implementation sketch:

```rust
pub fn read_from(path: impl AsRef<Path>, start_offset: u64)
    -> Result<(Vec<serde_json::Value>, u64), InboxError>
{
    let path = path.as_ref();
    let bytes = match std::fs::read(path) {
        Ok(b) => b,
        Err(source) => return Err(InboxError::Io { path: path.to_path_buf(), source }),
    };
    let file_len = bytes.len() as u64;
    if start_offset >= file_len {
        return Ok((Vec::new(), file_len));
    }
    let start = start_offset as usize;

    // Snap forward to the next \n boundary if start is mid-line.
    // Convention: byte at position `start - 1` SHOULD be `\n` (the
    // cursor was advanced to a line boundary). If it isn't, find the
    // next \n and skip past it.
    let snapped = if start == 0 || bytes.get(start - 1) == Some(&b'\n') {
        start
    } else {
        match bytes[start..].iter().position(|&b| b == b'\n') {
            Some(off) => start + off + 1,
            None => return Ok((Vec::new(), file_len)),
        }
    };

    let slice = &bytes[snapped..];
    let file_ends_with_newline = bytes.last() == Some(&b'\n');

    let chunks: Vec<&[u8]> = slice.split(|&b| b == b'\n').collect();
    let mut out = Vec::new();
    let mut consumed_through = snapped as u64;
    let total = chunks.len();

    for (idx, chunk) in chunks.iter().enumerate() {
        let is_last = idx + 1 == total;
        if is_last {
            // partial tail handling — only complete (newline-terminated) lines advance the offset
            if file_ends_with_newline && chunk.is_empty() { /* clean */ }
            // else: tail tolerance, no offset advance
            continue;
        }
        match serde_json::from_slice::<serde_json::Value>(chunk) {
            Ok(value) => {
                out.push(value);
                consumed_through += (chunk.len() as u64) + 1; // +1 for \n
            }
            Err(source) => {
                let line_no = idx + 1;
                return Err(InboxError::CorruptLine { line_no, source });
            }
        }
    }
    Ok((out, consumed_through))
}
```

**2. Create `crates/famp/src/cli/await_cmd/mod.rs`:**

```rust
//! `famp await` — block until a new inbox entry arrives past the cursor.

use std::time::{Duration, Instant};

use famp_inbox::{read::read_from, InboxCursor};
use serde_json::json;

use crate::cli::error::CliError;
use crate::cli::{home, paths};

pub mod poll;

#[derive(clap::Args, Debug)]
pub struct AwaitArgs {
    #[arg(long, default_value = "30s")]
    pub timeout: String,
    #[arg(long)]
    pub task: Option<String>,
}

const POLL_INTERVAL: Duration = Duration::from_millis(250);

pub async fn run(args: AwaitArgs) -> Result<(), CliError> {
    let home = home::resolve_famp_home()?;
    run_at(&home, args).await
}

pub async fn run_at(home: &std::path::Path, args: AwaitArgs) -> Result<(), CliError> {
    let timeout = crate::cli::error::parse_duration(&args.timeout)?;  // or wherever you put it
    let inbox_path = paths::inbox_jsonl_path(home);
    let cursor = InboxCursor::at(paths::inbox_cursor_path(home));
    let deadline = Instant::now() + timeout;

    loop {
        let start = cursor.read().await?;
        let (values, new_end) = read_from(&inbox_path, start)
            .map_err(CliError::Inbox)?;

        if let Some((entry, advance_to)) = poll::find_match(&values, start, &args.task) {
            // Print structured output, advance cursor, exit.
            println!("{}", serde_json::to_string(&entry).unwrap_or_default());
            cursor.advance(advance_to).await?;
            return Ok(());
        }

        // No match. If the filter consumed-and-discarded everything,
        // advance past the read batch so we do not re-poll the same
        // bytes forever.
        if !values.is_empty() && args.task.is_some() {
            cursor.advance(new_end).await?;
        }

        if Instant::now() >= deadline {
            return Err(CliError::AwaitTimeout { timeout: args.timeout });
        }
        tokio::time::sleep(POLL_INTERVAL).await;
    }
}
```

```rust
pub fn find_match(
    entries: &[(Value, u64)],
    task_filter: &Option<String>,
) -> Option<(Value, u64)> {
    for (value, end_offset) in entries {
        // Extract task_id from the envelope JSON. Phase 2 inbox.jsonl
        // stores raw envelope bytes verbatim — top-level "task_id"
        // field per FAMP v0.5.1 envelope schema.
        let task_id = value.get("task_id").and_then(|v| v.as_str());
        let from = value.get("from").and_then(|v| v.as_str()).unwrap_or("");
        let class = value.get("class").and_then(|v| v.as_str()).unwrap_or("");
        let body = value.get("body").cloned().unwrap_or(Value::Null);

        if let Some(filter) = task_filter {
            if task_id != Some(filter.as_str()) { continue; }
        }

        let out = json!({
            "offset": end_offset,
            "task_id": task_id.unwrap_or(""),
            "from": from,
            "class": class,
            "body": body,
        });
        return Some((out, *end_offset));
    }
    None
}
```
